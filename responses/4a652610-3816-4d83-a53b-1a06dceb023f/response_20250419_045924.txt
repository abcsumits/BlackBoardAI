from manim import *

class Frame(Scene):
    def construct(self):
        self.camera.background_color = "#000000"
        install_command = Text("> pip install rich", color="#FFFFFF", font_size=48)
        install_command.move_to([0, 0, 0])
        self.play(Write(install_command))
        self.wait(1)
