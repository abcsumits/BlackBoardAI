from manim import *

class Frame(Scene):
    def construct(self):
        # Define colors
        background_color = "#1E1E1E" # Dark gray background
        text_color = "#FFBF00"      # Gold-like color for text
        glow_color = "#FFFFE0"      # Light yellow for subtle glow effect
        initial_rect_color = "#CCCCCC" # Light gray for the initial rectangle

        # Get frame dimensions
        frame_height = config.frame_height
        frame_width = config.frame_width

        # 1. Initial State: Rectangle on the right
        # Assume this rectangle exists from a previous segment. We create it here for the transition.
        initial_rect = Rectangle(
            width=frame_width / 2,
            height=frame_height,
            fill_color=initial_rect_color,
            fill_opacity=1,
            stroke_width=0
        )
        # Position it on the right half of the screen
        initial_rect.move_to(np.array([frame_width / 4, 0, 0]))
        self.add(initial_rect) # Add instantly, assuming it's the end state of previous segment

        # 2. Animation 1: Rectangle expands to fill the screen
        full_screen_rect = Rectangle(
            width=frame_width,
            height=frame_height,
            fill_color=background_color,
            fill_opacity=1,
            stroke_width=0
        )
        full_screen_rect.move_to(ORIGIN) # Center the full screen rectangle

        # The Transform replaces initial_rect with full_screen_rect
        self.play(Transform(initial_rect, full_screen_rect, run_time=1.0))

        # 3. Animation 2: Stylized Text "rich" appears with a subtle glow
        rich_text = Text(
            "rich",
            font_size=144,
            color=text_color,
            # Subtle glow effect using stroke
            stroke_color=glow_color,
            stroke_width=2.5 # Adjust stroke width for desired glow intensity
        )
        rich_text.move_to(ORIGIN) # Center the text

        # Animate the appearance of the text
        self.play(Write(rich_text, run_time=1.5))

        # Add a subtle animation to enhance the "glow" feel - a gentle pulse
        self.play(
            rich_text.animate.scale(1.05), # Scale up slightly
            rate_func=there_and_back,      # Scale back down
            run_time=1.5                   # Duration of the pulse
        )

        # Keep the final frame visible for a moment
        self.wait(1.5)
