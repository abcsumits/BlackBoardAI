```python
from manim import *
import math

class Frame(Scene):
    def construct(self):
        # Configuration
        bar_width = 8.0
        bar_height = 0.5
        bar_outline_color = "#FFFFFF"  # White outline
        fill_color = "#00FF00" # Green fill
        text_color = "#FFFFFF" # White text

        # Progress tracker (value from 0 to 1)
        progress = ValueTracker(0)

        # Progress bar outline
        outline_bar = Rectangle(
            width=bar_width,
            height=bar_height,
            color=bar_outline_color,
            stroke_width=2
        )
        outline_bar.move_to(ORIGIN) # Center the outline bar

        # Progress bar filling rectangle (initially zero width)
        # Start it aligned to the left edge of the outline
        fill_bar = Rectangle(
            width=0.001, # Start with minimal width to avoid issues
            height=bar_height,
            fill_color=fill_color,
            fill_opacity=1.0,
            stroke_width=0
        ).align_to(outline_bar, LEFT)

        # Updater function for the fill bar's width
        def fill_updater(mob):
            # Calculate current width based on progress tracker
            current_width = progress.get_value() * bar_width
            # Set the width, ensuring it's non-zero for visibility
            mob.stretch_to_fit_width(max(0.001, current_width), about_edge=LEFT)
            # Keep it aligned to the left of the outline
            mob.align_to(outline_bar, LEFT)

        # Apply the updater to the fill bar
        fill_bar.add_updater(fill_updater)

        # Percentage text positioned above the bar
        percentage_text = Text("0%", font_size=48, color=text_color)
        percentage_text.next_to(outline_bar, UP, buff=0.5)

        # Store initial position for the text updater
        initial_text_position = percentage_text.get_center()

        # Updater function for the percentage text
        def text_updater(mob):
            # Calculate percentage value
            value = int(math.floor(progress.get_value() * 100))
            # Update the text content and maintain position
            new_text = Text(f"{value}%", font_size=48, color=text_color)
            new_text.move_to(initial_text_position)
            mob.become(new_text)


        # Apply the updater to the percentage text
        percentage_text.add_updater(text_updater)

        # Group elements for centering
        # content_group = VGroup(percentage_text, outline_bar)
        # content_group.move_to(ORIGIN) # Center the whole group initially

        # Add elements to the scene
        self.play(Create(outline_bar), Write(percentage_text))
        self.add(fill_bar) # Add fill bar (it starts nearly invisible)

        # Animate the progress tracker from 0 to 1
        self.play(progress.animate.set_value(1), run_time=4, rate_func=linear)

        # Explicitly remove updaters after animation (good practice)
        fill_bar.remove_updater(fill_updater)
        percentage_text.remove_updater(text_updater)

        # Keep the final frame visible for a moment
        self.wait(1.5)
```
