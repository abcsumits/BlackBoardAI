```python
from manim import *

class Frame(Scene):
    def construct(self):
        # Context Text
        context_text = Text(
            "Rich makes creating clean readable tables incredibly simple.",
            font_size=28,
            color="#FFFFFF"
        )
        context_text.to_edge(UP, buff=1.0)
        self.play(Write(context_text))
        self.wait(0.5)

        # Table Data
        table_data = [
            ["Column A", "Column B"],
            ["Data One", "Data Two"],
            ["More Data", "Even More"]
        ]

        # Create Table Mobject
        # Manim Community v0.19.0 uses slightly different Table init args
        table = Table(
            table_data,
            include_header_line=True, # This adds a line under the first row
            line_config={"color": "#CCCCCC", "stroke_width": 2},
            element_to_mobject=Text, # Use Text mobject for entries
            element_to_mobject_config={"color": "#FFFFFF"} # Config for Text mobjects
        )
        table.scale(0.7)
        table.next_to(context_text, DOWN, buff=0.7)

        # Animate Table Creation
        # 1. Create table lines
        table_lines = table.get_lines()
        self.play(Create(table_lines))
        self.wait(0.5)

        # 2. Fade in Headers (entries in the first row)
        headers = table.get_entries_without_labels()[0]
        self.play(FadeIn(headers))
        self.wait(0.5)

        # 3. Fade in Data Row 1 (entries in the second row)
        row1 = table.get_entries_without_labels()[1]
        self.play(FadeIn(row1))
        self.wait(0.5)

        # 4. Fade in Data Row 2 (entries in the third row)
        row2 = table.get_entries_without_labels()[2]
        self.play(FadeIn(row2))
        self.wait(1.5) # Hold the final table longer
```
