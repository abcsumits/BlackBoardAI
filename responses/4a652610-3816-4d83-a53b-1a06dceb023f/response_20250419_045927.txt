from manim import *

class Frame(Scene):
    def construct(self):
        # Define elements from "previous segments" - scaled appropriately
        previous_text = Text("Rich Output", color='#FF0000', weight='BOLD').scale(1.1)
        previous_table = Table(
            [["Feature", "Example"],
             ["Color", "[bold red]Red[/]"],
             ["Style", "[italic]Italic[/]"]],
            include_outer_lines=True,
            line_config={"stroke_color": "#FFFFFF"} # Make table lines visible
        ).scale(0.6)
        # Progress bar representation
        bar_outline = Rectangle(width=6, height=0.4, stroke_color='#FFFFFF', stroke_width=2)
        # Fill the bar almost completely to represent progress
        bar_fill = Rectangle(width=5.8, height=0.4, fill_color='#44FF44', fill_opacity=1, stroke_width=0).align_to(bar_outline, LEFT).shift(RIGHT*0.1)
        previous_progress_bar = VGroup(bar_outline, bar_fill).scale(0.7)

        # Position elements centered for their brief appearance
        previous_text.move_to(ORIGIN)
        previous_table.move_to(ORIGIN)
        previous_progress_bar.move_to(ORIGIN)

        # Rapid sequence of FadeIn/FadeOut centered
        self.play(FadeIn(previous_text), run_time=0.35)
        self.play(FadeOut(previous_text), run_time=0.35)
        self.wait(0.05) # Short pause

        self.play(FadeIn(previous_table), run_time=0.35)
        self.play(FadeOut(previous_table), run_time=0.35)
        self.wait(0.05)

        self.play(FadeIn(previous_progress_bar), run_time=0.35)
        self.play(FadeOut(previous_progress_bar), run_time=0.35)
        self.wait(0.1) # Slightly longer pause before final text

        # Final text
        final_text = Text("Explore Python Rich!", color='#FFFF00').scale(1.2) # Yellow color
        final_text.move_to(ORIGIN) # Center the final text

        # Animate final text appearance using Write for better effect
        self.play(Write(final_text))
        self.wait(2) # Hold the final text on screen
