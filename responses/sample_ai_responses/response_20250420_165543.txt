from manim import *
import random

class Frame(Scene):
    def construct(self):
        # Colors
        ground_color = "#228B22"  # ForestGreen
        cube_color = "#A0522D"    # Sienna
        text_color = "#FFFFFF"    # White
        forum_border_color = "#808080" # Grey
        forum_fill_color = "#FFFFFF"   # White
        forum_text_color = "#000000" # Black
        cube_stroke_color = "#000000" # Black

        # Ground
        ground = Rectangle(width=10, height=1, color=ground_color, fill_opacity=1)
        ground.to_edge(DOWN, buff=1)

        # Initial Cubes (Assuming few exist from 'Cave Game' stage)
        initial_cubes = VGroup()
        initial_cube_positions = []
        for _ in range(2):
             cube = Cube(side_length=0.6, fill_color=cube_color, fill_opacity=1, stroke_color=cube_stroke_color, stroke_width=2)
             placed = False
             attempts = 0
             while not placed and attempts < 20:
                 attempts += 1
                 x_pos = random.uniform(-ground.width/2 + cube.width/2, ground.width/2 - cube.width/2)
                 target_pos = ground.get_top() + DOWN * cube.height/2 + RIGHT * x_pos
                 # Basic overlap check
                 overlapping = any(abs(target_pos[0] - p[0]) < cube.width for p in initial_cube_positions)
                 if not overlapping:
                     cube.move_to(target_pos)
                     initial_cubes.add(cube)
                     initial_cube_positions.append(target_pos)
                     placed = True

        # Initial Text
        cave_game_text = Text('Cave Game', color=text_color).scale(0.9)
        cave_game_text.next_to(ground, UP, buff=0.8)

        self.add(ground, initial_cubes, cave_game_text)
        self.wait(0.5)

        # Add more cubes randomly on the ground
        new_cubes = VGroup()
        cube_animations = []
        num_new_cubes = 8
        all_cube_positions = list(initial_cube_positions) # Copy initial positions

        for _ in range(num_new_cubes):
            new_cube = Cube(side_length=0.6, fill_color=cube_color, fill_opacity=1, stroke_color=cube_stroke_color, stroke_width=2)
            placed = False
            attempts = 0
            max_attempts = 50 # Prevent potential infinite loops

            while not placed and attempts < max_attempts:
                attempts += 1
                # Attempt to place on the ground
                x_pos = random.uniform(-ground.width/2 + new_cube.width/2, ground.width/2 - new_cube.width/2)
                target_pos = ground.get_top() + DOWN * new_cube.height/2 + RIGHT * x_pos

                # Simple overlap check with all existing cubes based on x-position
                overlapping = False
                for pos in all_cube_positions:
                    if abs(target_pos[0] - pos[0]) < new_cube.width * 0.95: # Check x-overlap primarily
                        overlapping = True
                        break

                if not overlapping:
                    placed = True
                    start_pos = target_pos + UP * 4 # Start above the scene
                    new_cube.move_to(start_pos)
                    new_cubes.add(new_cube)
                    all_cube_positions.append(target_pos) # Add new position for checking
                    cube_animations.append(new_cube.animate.move_to(target_pos))

        # Animate new cubes appearing and landing
        if cube_animations:
             self.play(AnimationGroup(*[Create(c) for c in new_cubes], lag_ratio=0.05))
             self.play(AnimationGroup(*cube_animations, lag_ratio=0.1))
        self.wait(0.5)
        all_cubes = VGroup(*initial_cubes, *new_cubes) # Combine all cubes

        # Change Text
        minecraft_alpha_text = Text('Minecraft Alpha', color=text_color).scale(0.9)
        minecraft_alpha_text.move_to(cave_game_text.get_center())
        self.play(ReplacementTransform(cave_game_text, minecraft_alpha_text))
        self.wait(0.5)

        # Forum Post Visual
        forum_post = Rectangle(width=4.5, height=2.8, color=forum_border_color, fill_color=forum_fill_color, fill_opacity=0.95, stroke_width=2)
        forum_post_title = Text("TIGSource Forums", color=forum_text_color, weight=BOLD).scale(0.5)
        forum_post_body = Text("Working on a little sandbox game...\nProgress update: Added blocks!\nThinking of calling it Minecraft?", color=forum_text_color).scale(0.4)
        forum_post_content = VGroup(forum_post_title, forum_post_body).arrange(DOWN, center=False, aligned_edge=LEFT, buff=0.2)
        forum_post_content.move_to(forum_post.get_center())
        forum_visual = VGroup(forum_post, forum_post_content)

        # Position forum post (try not to overlap main elements)
        forum_visual.scale(0.8)
        forum_visual.move_to(UP * 1.5 + RIGHT * 3.5)


        # Animate Forum Post
        self.play(FadeIn(forum_visual, scale=0.8)) # Fade in and slightly scale up
        self.wait(2.5) # Hold forum post
        self.play(FadeOut(forum_visual, scale=1.2)) # Fade out and slightly scale down
        self.wait(1) # Hold final scene
