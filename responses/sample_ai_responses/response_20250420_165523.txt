```python
from manim import *
import numpy as np

class Frame(Scene):
    def construct(self):
        # Define colors
        table_color = "#A0522D"  # Brownish color for the table
        grid_color = "#693C1F"   # Darker brown for grid lines
        creeper_color = "#2E7D32" # Green color for the Creeper face

        # Crafting Table UI
        table_size = 2.0
        crafting_rect = Rectangle(
            width=table_size,
            height=table_size,
            color=table_color,
            fill_color=table_color,
            fill_opacity=0.9,
            stroke_width=3
        )

        grid_lines = VGroup()
        step = table_size / 3
        # Vertical lines
        for i in range(1, 3):
            x_coord = -table_size / 2 + i * step
            line = Line(
                start=np.array([x_coord, -table_size / 2, 0]),
                end=np.array([x_coord, table_size / 2, 0]),
                color=grid_color,
                stroke_width=2.5
            )
            grid_lines.add(line)
        # Horizontal lines
        for i in range(1, 3):
            y_coord = -table_size / 2 + i * step
            line = Line(
                start=np.array([-table_size / 2, y_coord, 0]),
                end=np.array([table_size / 2, y_coord, 0]),
                color=grid_color,
                stroke_width=2.5
            )
            grid_lines.add(line)

        # Group crafting table elements
        crafting_table = VGroup(crafting_rect, grid_lines)
        # Position crafting table to the left
        crafting_table.shift(LEFT * 2.5)

        # Creeper Face Polygon (Simplified Shape)
        # Defines a blocky shape with a characteristic mouth cutout
        creeper_points = [
            np.array([-1, 1, 0]), np.array([1, 1, 0]),      # Top edge
            np.array([1, -0.5, 0]),                           # Right edge top part
            np.array([0.5, -0.5, 0]), np.array([0.5, -1, 0]), # Mouth right notch
            np.array([-0.5, -1, 0]), np.array([-0.5, -0.5, 0]),# Mouth left notch
            np.array([-1, -0.5, 0]),                          # Left edge top part
            np.array([-1, 1, 0])                              # Close loop
        ]

        creeper_face = Polygon(
            *creeper_points,
            color=creeper_color,
            fill_color=creeper_color,
            fill_opacity=1,
            stroke_width=0  # No border for the filled polygon
        )
        creeper_face.scale(0.9)  # Adjust size if needed
        # Position creeper face to the right
        creeper_face.shift(RIGHT * 2.5)

        # Animations
        self.play(Create(crafting_table), run_time=1.5)
        self.wait(0.5)
        self.play(FadeIn(creeper_face), run_time=1.0)

        # Keep the final scene visible for a moment
        self.wait(1)
```
