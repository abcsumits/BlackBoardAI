from manim import *
import random

class Frame(Scene):
    def construct(self):
        # Config
        ground_color = "#228B22" # Green
        cube_color = "#A0522D"   # Brown
        text_color = "#FFFFFF"   # White
        forum_bg_color = "#444444" # Dark Grey
        forum_text_color = "#FFFFFF" # White
        cube_size = 0.5

        # 1. Initial Setup (Ground, initial cubes, initial text)
        ground = Rectangle(width=7, height=1, fill_color=ground_color, fill_opacity=1, stroke_width=0)
        ground.to_edge(DOWN, buff=1.0)
        
        initial_cubes = VGroup()
        # Place initial cubes on the ground
        initial_positions = [
            ground.get_center() + UP * (ground.height/2 + cube_size/2) + LEFT * 1.0,
            ground.get_center() + UP * (ground.height/2 + cube_size/2),
            ground.get_center() + UP * (ground.height/2 + cube_size/2) + RIGHT * 1.0,
        ]

        for pos in initial_positions:
             cube = Cube(side_length=cube_size, fill_color=cube_color, fill_opacity=1, stroke_color="#000000", resolution=(6,6))
             cube.move_to(pos)
             initial_cubes.add(cube)

        initial_text = Text('Cave Game', color=text_color).scale(0.8)
        initial_text.next_to(ground, UP, buff=1.0) # Position relative to ground

        self.play(Create(ground))
        self.play(Create(initial_cubes), Write(initial_text))
        self.wait(0.5)

        # 2. Add More Cubes - Stack them randomly
        new_cubes = VGroup()
        animations = []
        
        # Define potential base surfaces for stacking (ground top + initial cube tops)
        stack_bases = [ground.get_top()] + [c.get_top() for c in initial_cubes]
        
        # Store occupied positions to avoid direct overlap (simple check)
        occupied_coords = set([(round(p[0], 1), round(p[1], 1)) for p in initial_positions])

        for _ in range(6): # Add several more cubes
            placed = False
            attempts = 0
            while not placed and attempts < 20: # Try a few times to find a non-overlapping spot
                attempts += 1
                # Pick a random base surface
                base_point = random.choice(stack_bases)
                
                # Calculate potential position slightly offset horizontally
                target_x = base_point[0] + random.uniform(-cube_size*1.5, cube_size*1.5)
                target_y = base_point[1] + cube_size / 2.0 # Place directly on top vertically
                
                # Simple check for occupied space
                coord_key = (round(target_x, 1), round(target_y, 1))
                if coord_key not in occupied_coords:
                     # Check if it's within reasonable horizontal bounds (roughly ground width)
                     if abs(target_x - ground.get_center()[0]) < ground.width / 2.0 - cube_size/2 :
                        new_cube = Cube(side_length=cube_size, fill_color=cube_color, fill_opacity=1, stroke_color="#000000", resolution=(6,6))
                        target_pos = np.array([target_x, target_y, 0])
                        new_cube.move_to(target_pos)
                        
                        new_cubes.add(new_cube)
                        animations.append(Create(new_cube))
                        
                        # Add this cube's top as a potential base
                        stack_bases.append(new_cube.get_top())
                        occupied_coords.add(coord_key)
                        placed = True

        if animations:
            self.play(AnimationGroup(*animations, lag_ratio=0.15))
        self.wait(0.5)
        
        # 3. Text Change
        final_text = Text('Minecraft Alpha', color=text_color).scale(0.8)
        final_text.move_to(initial_text.get_center()) # Keep same position
        self.play(Transform(initial_text, final_text)) # initial_text now holds final_text's content
        self.wait(0.5)

        # 4. Forum Post Fade In/Out
        forum_bg = RoundedRectangle(corner_radius=0.1, width=3.5, height=2.0, color=forum_bg_color, fill_color=forum_bg_color, fill_opacity=0.9, stroke_width=1, stroke_color=text_color)
        forum_title = Text("Forum: Progress", color=forum_text_color, weight=BOLD).scale(0.4)
        forum_body = Text("Shared updates...\nStacking blocks works!", color=forum_text_color).scale(0.35)
        forum_content = VGroup(forum_title, forum_body).arrange(DOWN, buff=0.15, center=False, aligned_edge=LEFT)
        forum_content.move_to(forum_bg.get_center())
        
        forum_post = VGroup(forum_bg, forum_content).scale(0.9)
        # Position the forum post away from the main elements, e.g., top right
        forum_post.to_corner(UP + RIGHT, buff=0.5)

        self.play(FadeIn(forum_post), run_time=0.75)
        self.wait(2.0) # Display duration
        self.play(FadeOut(forum_post), run_time=0.75)

        # Ensure final frame isn't empty
        self.wait(1)
