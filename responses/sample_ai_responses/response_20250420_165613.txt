from manim import *

class Frame(Scene):
    def create_stick_figure(self, height=1.0, color="#FFFFFF", stroke_width=4):
        # Calculate points relative to a potential center (ORIGIN)
        head_center_y = height * 0.35
        hip_center_y = -height * 0.15
        shoulder_y = height * 0.2
        foot_y = -height * 0.5

        # Head
        head_radius = height * 0.15
        head = Circle(radius=head_radius, color=color, stroke_width=stroke_width)
        head.move_to(UP * (head_center_y + head_radius))

        # Body
        body = Line(
            UP * head_center_y,
            UP * hip_center_y,
            color=color,
            stroke_width=stroke_width
        )

        # Arms
        arm_length = height * 0.25
        shoulder_width = height*0.05
        left_arm = Line(
            UP * shoulder_y + LEFT*shoulder_width,
            UP * (shoulder_y - 0.1*height) + LEFT * (arm_length + shoulder_width),
            color=color,
            stroke_width=stroke_width
        )
        right_arm = Line(
            UP * shoulder_y + RIGHT*shoulder_width,
            UP * (shoulder_y - 0.1*height) + RIGHT * (arm_length + shoulder_width),
            color=color,
            stroke_width=stroke_width
        )

        # Legs
        leg_spread = height * 0.15
        left_leg = Line(
            UP * hip_center_y,
            UP * foot_y + LEFT * leg_spread,
            color=color,
            stroke_width=stroke_width
        )
        right_leg = Line(
            UP * hip_center_y,
            UP * foot_y + RIGHT * leg_spread,
            color=color,
            stroke_width=stroke_width
        )

        figure = VGroup(head, body, left_arm, right_arm, left_leg, right_leg)
        # Center the figure roughly around the hip/body center
        figure.move_to(ORIGIN)
        return figure

    def construct(self):
        self.camera.background_color = "#000000"
        # 1. Initial Question Mark
        question_mark = Text("?", font_size=350, color="#FFFFFF", weight=BOLD)
        question_mark.move_to(ORIGIN)

        self.play(Write(question_mark), run_time=1.5)
        self.wait(0.5)

        # 2. Create Stick Figures with varying sizes
        figure_color = "#FFFFFF"
        figure_stroke_width = 6 # Slightly thicker lines

        stick_figure1 = self.create_stick_figure(height=1.8, color=figure_color, stroke_width=figure_stroke_width)
        stick_figure2 = self.create_stick_figure(height=1.2, color=figure_color, stroke_width=figure_stroke_width)
        stick_figure3 = self.create_stick_figure(height=2.2, color=figure_color, stroke_width=figure_stroke_width)
        stick_figure4 = self.create_stick_figure(height=1.5, color=figure_color, stroke_width=figure_stroke_width)
        stick_figure5 = self.create_stick_figure(height=1.0, color=figure_color, stroke_width=figure_stroke_width)


        # 3. Group and Arrange Stick Figures
        # Order them somewhat aesthetically by size
        stick_figures_group = VGroup(
            stick_figure5, # Smallest
            stick_figure2,
            stick_figure4,
            stick_figure1,
            stick_figure3 # Tallest
        ).arrange(RIGHT, buff=0.6) # Arrange horizontally with buffer

        stick_figures_group.scale(0.9) # Scale group slightly to ensure it fits well
        stick_figures_group.move_to(ORIGIN) # Center the entire group


        # 4. Transformation
        # Use FadeOut and FadeIn for a cleaner visual change than direct Transform
        self.play(
             FadeOut(question_mark, scale=0.5),
             FadeIn(stick_figures_group, scale=1.5),
             run_time=2.0
         )

        # Keep figures on screen
        self.wait(2)
