```python
from manim import *

class Frame(Scene):
    def construct(self):
        # Define colors using hex codes
        yellow_color = "#FFFF00"
        green_color = "#00FF00"

        # Define text objects
        minecon_text = Text('MineCon 2011', color=yellow_color)
        minecraft_version_text = Text('Minecraft 1.0', color=green_color)

        # Position text objects (default center)
        minecon_text.move_to(ORIGIN)
        minecraft_version_text.move_to(ORIGIN)

        # Animation sequence
        self.play(FadeIn(minecon_text))
        self.wait(1)
        self.play(FadeOut(minecon_text))
        self.wait(0.5)
        self.play(FadeIn(minecraft_version_text))
        self.wait(2) # Keep the final text on screen
```
