```python
from manim import *

class Frame(Scene):
    def construct(self):
        # Define Colors
        notch_blue = "#0078D4"
        mojang_red = "#E81123"
        ms_red = "#F25022"
        ms_green = "#7FBA00"
        ms_blue = "#00A4EF"
        ms_yellow = "#FFB900"

        # Define visual positions
        visual_center = UP * 1.0

        # Create Notch Circle
        notch_circle = Circle(color=notch_blue, fill_opacity=1).scale(0.8)
        notch_circle.move_to(visual_center)

        # Create Mojang Polygon (using a simple shape)
        # A regular pentagon for visual distinction
        mojang_poly = RegularPolygon(n=5, color=mojang_red, fill_opacity=1).scale(1.0)
        mojang_poly.move_to(visual_center)

        # Create Microsoft Logo (Group of 4 Squares)
        s = 0.7  # side length of small squares
        gap = 0.05 # gap between squares
        sq_ul = Square(color=ms_red, fill_opacity=1, side_length=s).move_to(np.array([-s/2 - gap/2, s/2 + gap/2, 0]))
        sq_ur = Square(color=ms_green, fill_opacity=1, side_length=s).move_to(np.array([s/2 + gap/2, s/2 + gap/2, 0]))
        sq_dl = Square(color=ms_blue, fill_opacity=1, side_length=s).move_to(np.array([-s/2 - gap/2, -s/2 - gap/2, 0]))
        sq_dr = Square(color=ms_yellow, fill_opacity=1, side_length=s).move_to(np.array([s/2 + gap/2, -s/2 - gap/2, 0]))
        microsoft_logo = VGroup(sq_ul, sq_ur, sq_dl, sq_dr).scale(1.3) # Scale the whole logo group
        microsoft_logo.move_to(visual_center)

        # Create Text elements
        text_notch_departs = Text(
            "Notch departed, feeling the pressure...",
            font_size=28,
            line_spacing=0.8
        ).to_edge(DOWN, buff=1.0)

        text_microsoft_era = Text(
            "Under Microsoft, Mojang Studios continued development.",
            font_size=28,
            line_spacing=0.8
        ).to_edge(DOWN, buff=1.0)

        # --- Animation Sequence ---

        # Show Notch circle
        self.play(FadeIn(notch_circle, scale=0.5))
        self.wait(1)

        # Fade out Notch circle
        self.play(FadeOut(notch_circle))
        self.wait(0.5)

        # Show Mojang polygon
        self.play(FadeIn(mojang_poly, scale=0.8))
        self.play(Write(text_notch_departs))
        self.wait(1.5)

        # Transform Mojang polygon into Microsoft logo and update text
        self.play(
            Transform(mojang_poly, microsoft_logo),
            ReplacementTransform(text_notch_departs, text_microsoft_era),
            run_time=2.0
        )
        self.wait(3) # Hold the final scene
```
