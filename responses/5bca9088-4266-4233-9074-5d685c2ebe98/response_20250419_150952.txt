```python
from manim import *
import numpy as np

class Frame(Scene):
    def construct(self):
        # 1. Create Axes
        axes = Axes(
            x_range=[0, 6, 1],  # Represents time progressing
            y_range=[0, 12, 2], # Represents activity level (arbitrary units)
            x_length=7,
            y_length=4,
            axis_config={"color": "#CCCCCC", "include_numbers": True},
            x_axis_config={"numbers_to_include": np.arange(1, 7)},
            y_axis_config={"numbers_to_include": np.arange(2, 13, 2)},
            tips=False,
        ).move_to(UP * 1.0) # Move axes slightly up

        axes_labels = axes.get_axis_labels(
            x_label=Text("Time", font_size=24, color="#FFFFFF"),
            y_label=Text("Activity", font_size=24, color="#FFFFFF")
        )

        # 2. Define and create the plot showing sharp decrease
        def activity_func(x):
            # Exponential decay function: starts high, decreases sharply
            # Adjusted parameters for visual appearance
            return 10 * np.exp(-x * 0.8) + 1

        # Ensure plot domain matches axes x_range start and end
        graph = axes.plot(activity_func, color="#5dade2", x_range=[0.1, 5.5]) # Light Blue color

        # 3. Create the context text
        context_text = Text(
            "Like many community projects for older platforms,\nMCP server activity has significantly decreased.",
            font_size=28,
            color="#FFFFFF",
            line_spacing=1
        ).next_to(axes, DOWN, buff=0.7) # Place text below the axes


        # 4. Group graph elements
        graph_group = VGroup(axes, axes_labels, graph)

        # 5. Animate creation
        self.play(Create(axes), Write(axes_labels), run_time=1.5)
        self.play(Create(graph), run_time=2)
        self.play(Write(context_text), run_time=2)

        # 6. Hold the final frame
        self.wait(2)
```
