```python
from manim import *

class Frame(Scene):
    def construct(self):
        # Define colors
        primary_color = "#FFFFFF" # White
        secondary_color = "#CCCCCC" # Light gray

        # Create the main text
        main_text = Text("MCP Server", font_size=60, color=primary_color)

        # Create the subtext
        sub_text = Text("Unofficial Community Project", font_size=36, color=secondary_color)

        # Position the texts vertically
        main_text.shift(UP * 0.5)
        sub_text.next_to(main_text, DOWN, buff=0.4)

        # Group the texts
        text_group = VGroup(main_text, sub_text)

        # Center the group on the screen
        text_group.move_to([0,0,0])

        # Animate the appearance of the main text
        self.play(Write(main_text), run_time=1.5)
        self.wait(0.5)

        # Animate the appearance of the subtext
        self.play(Write(sub_text), run_time=1.5)

        # Hold the final scene
        self.wait(2)
```
