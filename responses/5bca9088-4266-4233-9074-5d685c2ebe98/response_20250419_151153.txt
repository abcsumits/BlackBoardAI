```python
from manim import *

class Frame(Scene):
    def construct(self):
        # Configuration
        question_mark_color = "#FFFFFF"
        controller_body_color = "#CCCCCC"
        controller_detail_color = "#333333"
        button_a_color = "#E74C3C" # Reddish
        button_b_color = "#2ECC71" # Greenish
        title_color = "#FFFFFF"

        # Create Controller Icon
        body = Rectangle(width=2.0, height=1.0, color=controller_body_color, fill_opacity=1, stroke_width=0)
        dpad_h = Rectangle(width=0.6, height=0.2, color=controller_detail_color, fill_opacity=1, stroke_width=0)
        dpad_v = Rectangle(width=0.2, height=0.6, color=controller_detail_color, fill_opacity=1, stroke_width=0)
        dpad = VGroup(dpad_h, dpad_v).move_to(body.get_left() + RIGHT * 0.4)

        button_a = Circle(radius=0.15, color=button_a_color, fill_opacity=1, stroke_width=0).move_to(body.get_right() + LEFT * 0.6 + UP * 0.1)
        button_b = Circle(radius=0.15, color=button_b_color, fill_opacity=1, stroke_width=0).move_to(body.get_right() + LEFT * 0.3 + DOWN * 0.1)

        controller_icon = VGroup(body, dpad, button_a, button_b)

        # Create Question Mark
        question_mark = Text("?", font_size=144, color=question_mark_color)

        # Group Icon and Question Mark
        icon_group = VGroup(controller_icon, question_mark).arrange(RIGHT, buff=1.5)
        icon_group.move_to(ORIGIN + DOWN * 0.5) # Position slightly lower

        # Create Title Text
        title = Text("What is MCP Server?", font_size=60, color=title_color)
        title.next_to(icon_group, UP, buff=1.0) # Position above the icons

        # Animations
        self.play(Create(icon_group))
        self.play(Write(title))

        # Keep elements on screen
        self.wait(3)
```
