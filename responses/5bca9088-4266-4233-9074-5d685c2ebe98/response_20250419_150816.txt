from manim import Scene, Dot, Text, Write, Create, VGroup, UP, RIGHT, WHITE

class Frame(Scene):
    def construct(self):
        # Define colors using hex codes
        text_color = "#FFFFFF"
        dot_color = "#FFFFFF"
        
        # Create the second bullet point dot
        bullet_dot = Dot(color=dot_color)
        
        # Create the text for the second bullet point
        bullet_text = Text(
            "Custom Worlds: Play Unique Maps", 
            color=text_color, 
            font_size=36
        )
        
        # Group the dot and text
        bullet_group = VGroup(bullet_dot, bullet_text)
        
        # Arrange them horizontally with some buffer
        bullet_group.arrange(RIGHT, buff=0.2)
        
        # Position the group slightly above the center
        # (Assuming it's part of a list starting near the center)
        bullet_group.shift(UP * 0.5) 
        
        # Animate the appearance of the dot
        self.play(Create(bullet_dot))
        
        # Animate the appearance of the text
        self.play(Write(bullet_text))
        
        # Hold the final frame
        self.wait(2)
