from manim import *
import numpy as np

class Frame(Scene):
    def construct(self):
        # Define colors using hex codes
        book_color = "#A0522D"  # Sienna (for spine)
        page_color = "#F5F5DC"  # Beige
        diagram_color = "#808080" # Grey
        bulb_body_color = "#D3D3D3" # Light Grey (off)
        bulb_glow_color = "#FFFFE0" # Light Yellow (on)
        ray_color = "#FFD700"    # Gold/Yellow
        filament_color = "#404040" # Dark Grey

        # Define positions using numpy arrays
        book_center_pos = np.array([-2.5, 0, 0])
        lightbulb_pos = np.array([2.5, 0.5, 0])
        diagram_pos_offset = np.array([0, 0.5, 0]) # Relative to right page center

        # Book Representation
        page_width = 2.5
        page_height = 3.5
        left_page = Rectangle(
            width=page_width,
            height=page_height,
            color=page_color,
            fill_color=page_color,
            fill_opacity=1.0,
            stroke_color="#000000", # Black outline
            stroke_width=1
        ).move_to(book_center_pos + np.array([-page_width/2, 0, 0]))

        right_page = Rectangle(
            width=page_width,
            height=page_height,
            color=page_color,
            fill_color=page_color,
            fill_opacity=1.0,
            stroke_color="#000000", # Black outline
            stroke_width=1
        ).move_to(book_center_pos + np.array([page_width/2, 0, 0]))

        book_spine = Line(
            book_center_pos + np.array([0, -page_height/2, 0]),
            book_center_pos + np.array([0, page_height/2, 0]),
            color=book_color,
            stroke_width=8
        )
        book = VGroup(left_page, right_page, book_spine)

        # Diagram Placeholder (on the right page)
        diagram = Rectangle(
            width=1.5,
            height=1.0,
            color=diagram_color,
            fill_color=diagram_color,
            fill_opacity=0.7
        ).move_to(right_page.get_center() + diagram_pos_offset)

        # Lightbulb Icon
        bulb_radius = 0.5
        bulb_circle = Circle(
            radius=bulb_radius,
            color="#000000", # Black outline
            stroke_width=2,
            fill_color=bulb_body_color,
            fill_opacity=1.0
        ).move_to(lightbulb_pos)

        bulb_base = Rectangle(
             width=0.4,
             height=0.2,
             color="#000000", # Black outline
             stroke_width=2,
             fill_color=bulb_body_color,
             fill_opacity=1.0
        ).next_to(bulb_circle, DOWN, buff=0)

        # Simple filament representation
        filament = Line(
            bulb_circle.get_center() + np.array([-0.1, -0.1, 0]),
            bulb_circle.get_center() + np.array([0.1, 0.1, 0]),
            color=filament_color,
            stroke_width=2
        )
        filament2 = Line(
             bulb_circle.get_center() + np.array([0.1, -0.1, 0]),
             bulb_circle.get_center() + np.array([-0.1, 0.1, 0]),
             color=filament_color,
             stroke_width=2
         )

        lightbulb_shape = VGroup(bulb_circle, bulb_base, filament, filament2)

        # Rays
        rays = VGroup()
        num_rays = 8
        ray_length_inner = 0.1 # Gap between bulb and ray start
        ray_length_outer = 0.6 # Length of the ray itself
        for i in range(num_rays):
            angle = TAU * i / num_rays
            direction = np.array([np.cos(angle), np.sin(angle), 0])
            start_point = bulb_circle.get_center() + (bulb_radius + ray_length_inner) * direction
            end_point = start_point + ray_length_outer * direction
            ray = Line(start_point, end_point, color=ray_color, stroke_width=4)
            rays.add(ray)

        # --- Animation Sequence ---

        # 1. Show book and diagram
        self.play(FadeIn(book), FadeIn(diagram))
        self.wait(0.5)

        # 2. Show lightbulb shape (initially off)
        self.play(FadeIn(lightbulb_shape))
        self.wait(0.5)

        # 3. Lightbulb illuminates: Rays appear and bulb changes color
        self.play(
            LaggedStart(
                *[GrowFromCenter(ray) for ray in rays],
                lag_ratio=0.1
            ),
            bulb_circle.animate.set_fill(bulb_glow_color, opacity=1.0),
            bulb_base.animate.set_fill(bulb_glow_color, opacity=1.0),
            # Optional: make filament disappear or change color slightly
            # filament.animate.set_color("#FFFFFF"),
            # filament2.animate.set_color("#FFFFFF")
        )

        # Hold the final scene for a while
        self.wait(2.5)
