```python
from manim import *
import math

class Frame(Scene):
    def construct(self):
        # 1. Define colors
        book_cover_color = "#8B4513"
        book_page_color = "#F5F5DC"
        pc_color = "#ADD8E6"
        console_color = "#90EE90"
        arrow_color = "#FFFFFF"
        text_color = "#000000"

        # 2. Create Book Components (Closed Position)
        page_width = 2.0
        page_height = 3.0
        spine_width = 0.1

        # Position the spine at the center initially for easier rotation setup
        spine = Rectangle(
            width=spine_width,
            height=page_height,
            color=book_cover_color,
            fill_color=book_cover_color,
            fill_opacity=1.0,
            stroke_width=1
        ).move_to(ORIGIN)

        # Create left page flat initially, aligned with spine's left edge
        left_page = Rectangle(
            width=page_width,
            height=page_height,
            color=book_cover_color,
            fill_color=book_cover_color,
            fill_opacity=1.0,
            stroke_width=1
        )
        left_page.next_to(spine, LEFT, buff=0)

        # Create right page flat initially, aligned with spine's right edge
        right_page = Rectangle(
            width=page_width,
            height=page_height,
            color=book_cover_color,
            fill_color=book_cover_color,
            fill_opacity=1.0,
            stroke_width=1
        )
        right_page.next_to(spine, RIGHT, buff=0)

        # Group the closed book
        closed_book = VGroup(left_page, spine, right_page)
        closed_book.move_to(ORIGIN) # Center the closed book on screen

        # 3. Animate Book Creation and Opening
        self.play(Create(closed_book), run_time=1.0)
        self.wait(0.5)

        # Define rotation points/axes based on the spine's current position
        left_pivot = spine.get_edge_center(LEFT)
        right_pivot = spine.get_edge_center(RIGHT)

        # Animate opening using Rotate
        self.play(
            Rotate(left_page, angle=-math.pi * 0.45, axis=UP, about_point=left_pivot),
            Rotate(right_page, angle=math.pi * 0.45, axis=UP, about_point=right_pivot),
            run_time=1.5
        )
        self.wait(0.2)

        # 4. Create and reveal inner pages and diagram
        # Create inner page rectangles slightly smaller and positioned slightly 'above' covers
        inner_left_page = Rectangle(
            width=page_width * 0.95,
            height=page_height * 0.95,
            color=book_page_color,
            fill_color=book_page_color,
            fill_opacity=1.0,
            stroke_width=0 # No stroke for inner pages
        ).move_to(left_page.get_center() + OUT*0.01) # Position relative to cover

        inner_right_page = Rectangle(
             width=page_width * 0.95,
            height=page_height * 0.95,
            color=book_page_color,
            fill_color=book_page_color,
            fill_opacity=1.0,
            stroke_width=0
        ).move_to(right_page.get_center() + OUT*0.01) # Position relative to cover

        # Rotate inner pages to match the opened covers
        inner_left_page.rotate(angle=-math.pi * 0.45, axis=UP, about_point=left_pivot)
        inner_right_page.rotate(angle=math.pi * 0.45, axis=UP, about_point=right_pivot)

        # Create the diagram elements
        pc_rect = Rectangle(
            width=1.2, height=0.8, color=pc_color, fill_color=pc_color, fill_opacity=0.9, stroke_width=1, stroke_color=text_color
            )
        pc_text = Text("PC", color=text_color).scale(0.4)
        # Position PC group relative to the center of the rotated inner left page
        pc_group = VGroup(pc_rect, pc_text).move_to(inner_left_page.get_center() + OUT*0.02)


        console_sq = Square(
            side_length=1.0, color=console_color, fill_color=console_color, fill_opacity=0.9, stroke_width=1, stroke_color=text_color
            )
        console_text = Text("Modded\nConsole", color=text_color, line_spacing=0.7).scale(0.3)
        # Position Console group relative to the center of the rotated inner right page
        console_group = VGroup(console_sq, console_text).move_to(inner_right_page.get_center() + OUT*0.02)


        # Reposition groups slightly for better layout within pages
        pc_group.shift(RIGHT * 0.1)
        console_group.shift(LEFT * 0.1)


        # Create arrow after potential positions are set
        arrow = Arrow(
            start=pc_group.get_edge_center(RIGHT),
            end=console_group.get_edge_center(LEFT),
            color=arrow_color,
            stroke_width=4,
            tip_length=0.2,
            buff=0.2
        ).shift(OUT*0.02) # Elevate arrow slightly

        # Animate the diagram appearance along with inner pages
        self.play(
            FadeIn(inner_left_page),
            FadeIn(inner_right_page),
            FadeIn(pc_group, shift=LEFT*0.1), # Slight shift animation
            FadeIn(console_group, shift=RIGHT*0.1), # Slight shift animation
            GrowArrow(arrow),
            run_time=1.5
        )

        # 5. Keep final state visible
        self.wait(3)
```
