```python
from manim import *

class Frame(Scene):
    def construct(self):
        # Define colors
        pc_color = "#3498DB" # Blue
        console_color = "#2ECC71" # Green
        arrow_color = "#F1C40F" # Yellow
        text_color = "#ECF0F1" # White/Light Gray

        # Create PC Rectangle
        pc_rect = Rectangle(width=3.0, height=2.0, color=pc_color, fill_opacity=0.5)
        pc_text = Text("PC", color=text_color).scale(0.7)
        pc_group = VGroup(pc_rect, pc_text).move_to([-4, 0, 0])

        # Create Modded Console Square
        console_sq = Square(side_length=2.0, color=console_color, fill_opacity=0.5)
        console_text = Text("MODDED\nConsole", color=text_color, line_spacing=0.7).scale(0.5)
        console_group = VGroup(console_sq, console_text).move_to([4, 0, 0])

        # Display PC and Console
        self.play(Create(pc_rect), Write(pc_text))
        self.play(Create(console_sq), Write(console_text))
        self.wait(0.5)

        # Create Arrow
        arrow_start = pc_rect.get_right() + RIGHT * 0.1
        arrow_end = console_sq.get_left() + LEFT * 0.1
        arrow = Arrow(start=arrow_start, end=arrow_end, buff=0.1, color=arrow_color)

        # Create Custom IP Text
        custom_ip_text = Text("Custom IP", color=text_color).scale(0.6)
        custom_ip_text.next_to(arrow.get_center(), UP, buff=0.2)


        # Animate Arrow and Text
        self.play(GrowArrow(arrow))
        self.play(AddTextLetterByLetter(custom_ip_text))

        # Hold the final scene
        self.wait(2)

```
