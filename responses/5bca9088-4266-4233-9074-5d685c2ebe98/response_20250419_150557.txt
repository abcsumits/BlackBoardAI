```python
from manim import *

class Frame(Scene):
    def construct(self):
        # Define colors using hex codes
        console_color = "#4CAF50" # Greenish
        friend_color = "#2196F3"  # Blue
        text_color = "#FFFFFF"    # White
        arrow_color = "#FFFFFF"   # White

        # Create the central "Console" square and its label
        console_label = Text("Console", color=text_color, font_size=32)
        console_square = Square(color=console_color, side_length=2.0)
        console_group = VGroup(console_square, console_label).move_to(ORIGIN + DOWN * 1.5) # Position slightly lower center

        # Create the first "Friend" square and its label
        friend1_label = Text("Friend", color=text_color, font_size=32)
        friend1_square = Square(color=friend_color, side_length=2.0)
        friend1_group = VGroup(friend1_square, friend1_label).next_to(console_group, UP + LEFT, buff=1.5) # Position top-left relative to console

        # Create the second "Friend" square and its label
        friend2_label = Text("Friend", color=text_color, font_size=32)
        friend2_square = Square(color=friend_color, side_length=2.0)
        friend2_group = VGroup(friend2_square, friend2_label).next_to(console_group, UP + RIGHT, buff=1.5) # Position top-right relative to console

        # Create arrows connecting Console to Friends
        # Adjust start/end points slightly away from the squares' edges
        arrow1 = Arrow(
            start=console_square.get_top() + UP * 0.1,
            end=friend1_square.get_bottom() + DOWN * 0.1,
            color=arrow_color,
            buff=0.1 # Keep arrow tips slightly away from the target edge
        )
        arrow2 = Arrow(
            start=console_square.get_top() + UP * 0.1,
            end=friend2_square.get_bottom() + DOWN * 0.1,
            color=arrow_color,
            buff=0.1 # Keep arrow tips slightly away from the target edge
        )

        # Animate the creation of the elements
        self.play(
            Create(console_square),
            Write(console_label),
            run_time=1.0
        )
        self.play(
            Create(friend1_square),
            Write(friend1_label),
            Create(friend2_square),
            Write(friend2_label),
            run_time=1.5
        )
        self.play(
            GrowArrow(arrow1),
            GrowArrow(arrow2),
            run_time=1.5
        )

        # Hold the final scene for a moment
        self.wait(2)
```
