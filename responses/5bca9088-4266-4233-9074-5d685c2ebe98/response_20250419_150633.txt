```python
from manim import *

class Frame(Scene):
    def construct(self):
        # Configuration
        mcp_text_color = "#FFFF00"  # Bright Yellow
        arrow_color = "#FFFFFF"     # White
        context_color = "#FFFFFF"   # White
        dot_color = "#40C4FF"       # Light Blue

        # Context Text
        context_text = Text(
            "this is where mcp server comes into the picture.",
            font_size=28,
            color=context_color
        ).to_edge(UP)
        self.play(Write(context_text))
        self.wait(1)

        # MCP Server Text with Glow Effect
        mcp_text = Text(
            'MCP Server',
            font_size=60,
            color=mcp_text_color,
            weight=BOLD
        ).shift(DOWN * 0.5)

        # Simple representation for "thought bubbles" source
        thought_source_location = LEFT * 4 + UP * 1
        source_dot = Dot(point=thought_source_location, color=dot_color, radius=0.15)

        # Arrow
        arrow = Arrow(
            start=source_dot.get_right() + RIGHT * 0.1, # Start slightly right of the dot
            end=mcp_text.get_left() + LEFT * 0.1,      # End slightly left of the text
            color=arrow_color,
            buff=0.1
        )

        # Animations
        self.play(Create(source_dot), run_time=0.5)
        self.play(Write(mcp_text), run_time=1.5)
        self.play(GrowArrow(arrow), run_time=1)

        # Keep elements on screen
        self.wait(2)
```
