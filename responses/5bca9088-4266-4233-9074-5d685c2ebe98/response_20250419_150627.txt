```python
from manim import *
import numpy as np

class Frame(Scene):
    def construct(self):
        # Define Colors using hex codes
        CONSOLE_COLOR = "#4A4A4A" # Dark grey
        BUBBLE_COLOR = "#F0F0F0" # Light grey
        ICON_COLOR_1 = "#FF8C42" # Orange (Minigames)
        ICON_COLOR_2 = "#4CAF50" # Green (Tools)
        ICON_COLOR_3 = "#2196F3" # Blue (Worlds)
        OUTLINE_COLOR = "#DDDDDD" # Lighter Grey outline/connector color

        # 1. Console Representation (Square)
        # Positioned towards the bottom center
        console = Square(side_length=2.0, color=CONSOLE_COLOR, fill_opacity=1.0)
        console.shift(DOWN * 2.0)

        # 2. Thought Bubbles (Ellipses)
        bubble_scale = 0.75 # Adjusted scale for better fit
        # Create three bubbles
        bubble1 = Ellipse(width=2.0, height=1.2, color=OUTLINE_COLOR, fill_color=BUBBLE_COLOR, fill_opacity=0.9).scale(bubble_scale)
        bubble2 = Ellipse(width=2.0, height=1.2, color=OUTLINE_COLOR, fill_color=BUBBLE_COLOR, fill_opacity=0.9).scale(bubble_scale)
        bubble3 = Ellipse(width=2.0, height=1.2, color=OUTLINE_COLOR, fill_color=BUBBLE_COLOR, fill_opacity=0.9).scale(bubble_scale)

        # Position bubbles above console, arranged horizontally without overlap
        bubble_group = VGroup(bubble1, bubble2, bubble3).arrange(RIGHT, buff=0.4) # Use VGroup for collective positioning
        bubble_group.next_to(console, UP, buff=0.3) # Position above console

        # Connecting dots from console to bubbles (visual flourish)
        start_point = console.get_top() + UP * 0.05 # Start just above console
        conn1 = Dot(start_point, color=OUTLINE_COLOR, radius=0.07)
        conn2 = Dot(start_point + UP*0.15, color=OUTLINE_COLOR, radius=0.05)
        conn3 = Dot(start_point + UP*0.30, color=OUTLINE_COLOR, radius=0.03) # Top dot where bubbles emerge
        connectors = VGroup(conn1, conn2, conn3)

        # 3. Icons (Simple Shapes)
        # Position icons inside their respective bubbles
        # Icon 1: Minigame (Star)
        icon1 = Star(n=5, outer_radius=0.28, inner_radius=0.14, color=ICON_COLOR_1, fill_opacity=1.0, stroke_width=0)
        icon1.move_to(bubble1.get_center())

        # Icon 2: Tool (Hammer)
        hammer_head = Rectangle(height=0.25, width=0.35, color=ICON_COLOR_2, fill_opacity=1.0, stroke_width=0)
        hammer_handle = Rectangle(height=0.40, width=0.1, color=ICON_COLOR_2, fill_opacity=1.0, stroke_width=0)
        hammer_handle.next_to(hammer_head, DOWN, buff=0)
        icon2 = VGroup(hammer_head, hammer_handle) # Group parts
        icon2.move_to(bubble2.get_center())

        # Icon 3: Custom World (Planet with Ring)
        planet_center = bubble3.get_center()
        planet = Circle(radius=0.18, color=ICON_COLOR_3, fill_opacity=1.0, stroke_width=0).move_to(planet_center)
        ring = Ellipse(width=0.55, height=0.14, color=ICON_COLOR_3, fill_opacity=1.0, stroke_width=0)
        ring.rotate(np.pi / 7) # Use numpy.pi
        ring.move_to(planet_center)
        icon3 = VGroup(planet, ring) # Group parts

        # --- Animation Sequence ---

        # Create Console
        self.play(Create(console), run_time=1.0)
        self.wait(0.1)

        # Animate connector dots appearing upwards
        self.play(GrowFromCenter(conn1), run_time=0.15)
        self.play(GrowFromCenter(conn2), run_time=0.15)
        self.play(GrowFromCenter(conn3), run_time=0.15)
        self.wait(0.1)

        # Create Bubbles growing from the top connector dot
        self.play(
            GrowFromPoint(bubble1, conn3.get_center()),
            GrowFromPoint(bubble2, conn3.get_center()),
            GrowFromPoint(bubble3, conn3.get_center()),
            run_time=1.0
        )
        self.wait(0.2)

        # Create Icons inside the bubbles
        self.play(
            SpinIn(icon1, angle=np.pi*2), # Use numpy.pi
            DrawBorderThenFill(icon2),
            FadeIn(icon3, scale=0.4),
            run_time=1.2
        )
        self.wait(0.3)

        # Add subtle animation to icons (bounce/wiggle effect)
        self.play(
            # Icon 1 bounces up slightly
            icon1.animate.shift(UP*0.08),
            # Icon 2 wiggles slightly
            icon2.animate(rate_func=wiggle).rotate(0.15, about_point=icon2.get_bottom()),
            # Icon 3 scales up slightly
            icon3.animate.scale(1.1),
            run_time=0.75 # First part of the animation
        )
        self.play(
             # Icon 1 bounces back down
            icon1.animate.shift(DOWN*0.08),
             # Icon 2 wiggles back (rotate back to original position)
            icon2.animate.rotate(-0.15, about_point=icon2.get_bottom()),
            # Icon 3 scales back down
            icon3.animate.scale(1/1.1),
            run_time=0.75 # Second part of the animation
        )

        # Hold the final frame for a moment before ending
        self.wait(2.5)
```
