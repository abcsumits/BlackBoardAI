from manim import *

class Frame(Scene):
    def construct(self):
        # Define text content for the 'boring' output
        plain_text_content = """
[INFO] Processing data...
[WARN] Skipping optional file.
[DEBUG] Value set to 10.
[ERROR] Critical failure!
        """.strip()

        # Define text content for the 'attractive' output using MarkupText for colors
        rich_text_content = """
<span foreground='#5FD7FF'>[INFO]</span> Processing data...
<span foreground='#FFFF87'>[WARN]</span> Skipping optional file.
<span foreground='#87FF87'>[DEBUG]</span> Value set to 10.
<span foreground='#FF8787' weight='bold'>[ERROR]</span> Critical failure!
        """.strip()

        # Create the plain grey text object
        text1 = Text(plain_text_content, color='#AAAAAA', font='Monospace', font_size=24)
        # Create a rectangle around the plain text
        rect1 = Rectangle(
            width=text1.width + 0.5,
            height=text1.height + 0.5,
            color='#FFFFFF',  # White border
            stroke_width=2
        ).round_corners(0.1) # Slightly rounded corners
        # Group the plain text and its rectangle
        group1 = VGroup(rect1, text1)
        # Position the first group on the left
        group1.move_to(LEFT * 3.5)

        # Create the colorful rich text object
        text2 = MarkupText(rich_text_content, font='Monospace', font_size=24)
        # Create a rectangle around the rich text
        rect2 = Rectangle(
            width=text2.width + 0.5,
            height=text2.height + 0.5,
            color='#FFFFFF',  # White border
            stroke_width=2
        ).round_corners(0.1) # Slightly rounded corners
        # Group the rich text and its rectangle
        group2 = VGroup(rect2, text2)
        # Position the second group on the right
        group2.move_to(RIGHT * 3.5)

        # Animate the appearance of both groups side-by-side
        self.play(Create(group1), Create(group2), run_time=1.5)
        self.wait(1)

        # Animate the fading out of the plain grey text group
        self.play(FadeOut(group1), run_time=1)
        self.wait(0.5)

        # Animate the colorful text group moving to the center and growing slightly
        self.play(group2.animate.move_to(ORIGIN).scale(1.1), run_time=1.5)
        
        # Hold the final frame for a couple of seconds
        self.wait(2)
