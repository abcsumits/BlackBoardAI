```python
from manim import *

class Frame(Scene):
    def construct(self):
        # Define colors and font
        color_default = '#FFFFFF'
        color_string = '#87ff00' # Green for strings
        color_number = '#00bfff' # Blue for numbers
        color_error = '#FF0000'  # Red for error
        color_status = '#0000FF' # Blue for status
        font_name = 'monospace'

        # 1. Initial Data Structure (as individual Text objects)
        t_brace_o = Text("{", font=font_name, color=color_default)
        t_key1 = Text("'name'", font=font_name, color=color_default)
        t_colon1 = Text(":", font=font_name, color=color_default)
        t_val1 = Text("'gadget'", font=font_name, color=color_default)
        t_comma = Text(",", font=font_name, color=color_default)
        t_key2 = Text("'id'", font=font_name, color=color_default)
        t_colon2 = Text(":", font=font_name, color=color_default)
        t_val2 = Text("123", font=font_name, color=color_default)
        t_brace_c = Text("}", font=font_name, color=color_default)

        # Arrange into a VGroup
        initial_dict_group = VGroup(
            t_brace_o, t_key1, t_colon1, t_val1, t_comma,
            t_key2, t_colon2, t_val2, t_brace_c
        ).arrange(RIGHT, buff=0.1).shift(UP * 1.5)

        # Display Initial structure
        self.play(Write(initial_dict_group))
        self.wait(1)

        # 2. Formatted Data Structure (Targets for transformation)
        # Create new Text objects with target colors
        ft_key1 = Text("'name'", font=font_name, color=color_string)
        ft_val1 = Text("'gadget'", font=font_name, color=color_string)
        ft_key2 = Text("'id'", font=font_name, color=color_string) # Keys are also strings here
        ft_val2 = Text("123", font=font_name, color=color_number)

        # Align target objects to the initial objects' positions
        ft_key1.move_to(t_key1)
        ft_val1.move_to(t_val1)
        ft_key2.move_to(t_key2)
        ft_val2.move_to(t_val2)

        # 3. Transformation (Color change)
        # Transform the specific parts that change color
        self.play(
            Transform(t_key1, ft_key1),
            Transform(t_val1, ft_val1),
            Transform(t_key2, ft_key2),
            Transform(t_val2, ft_val2),
            run_time=1.5
        )
        # The initial_dict_group now holds the transformed elements visually
        self.wait(1)

        # 4. Markup Text
        error_text = Text(
            '[bold red]error[/bold red]',
            font=font_name,
            t2c={'[bold red]': color_error},
            t2w={'[bold red]': BOLD}
        )
        status_text = Text(
            '[italic blue]status[/italic blue]',
            font=font_name,
            t2c={'[italic blue]': color_status},
            t2s={'[italic blue]': ITALIC}
        )

        # Position markup text below the dictionary
        error_text.next_to(initial_dict_group, DOWN, buff=0.8)
        status_text.next_to(error_text, DOWN, buff=0.4)

        # 5. Fade In Markup Sequentially
        self.play(FadeIn(error_text))
        self.wait(0.5)
        self.play(FadeIn(status_text))

        # 6. End scene holding the final state
        self.wait(2)

```
