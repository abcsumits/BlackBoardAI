from manim import *

class Frame(Scene):
    def construct(self):
        # Colors
        target_color = "#FFFF00" # Yellow
        middle_color = "#00FFFF" # Cyan
        other_color = "#FFFFFF" # White
        pointer_color = "#FF00FF" # Magenta
        comparison_color = "#FFFFFF" # White

        # 1. Setup initial elements (simplified representation)
        target_text = Text("23", color=target_color)
        middle_text = Text("38", color=middle_color)
        right_text = Text("56", color=other_color)

        # Position elements horizontally
        # Using ORIGIN for centering reference instead of absolute CENTER
        elements_group = VGroup(target_text, middle_text, right_text).arrange(RIGHT, buff=1.5)
        elements_group.move_to(ORIGIN) # Place the group near the center

        # High pointer (simplified)
        high_label = Text("high", font_size=24, color=pointer_color)
        # Position arrow start relative to label, end relative to text
        high_arrow = Arrow(
            start=high_label.get_bottom() + DOWN * 0.1,
            end=right_text.get_top() + UP * 0.1,
            color=pointer_color,
            buff=0.1,
            stroke_width=4,
            max_tip_length_to_length_ratio=0.15,
            max_stroke_width_to_length_ratio=3
        )
        # Group label and arrow, position relative to the '56' text
        high_pointer = VGroup(high_label, high_arrow).arrange(DOWN, buff=0.1)
        high_pointer.next_to(right_text, UP, buff=0.5)


        self.play(
            Write(target_text),
            Write(middle_text),
            Write(right_text),
            Create(high_pointer)
        )
        self.wait(0.5)

        # 2. Show comparison
        # Create Tex objects for better number/symbol rendering
        comparison_target = Tex("23", color=target_color).scale(1.2)
        comparison_lt = Tex("<", color=comparison_color).scale(1.2)
        comparison_middle = Tex("38", color=middle_color).scale(1.2)

        # Arrange comparison elements and position below the main numbers
        comparison_group = VGroup(comparison_target, comparison_lt, comparison_middle).arrange(RIGHT, buff=0.4)
        comparison_group.next_to(elements_group, DOWN, buff=1.2)

        # Animate the appearance of the comparison
        self.play(Write(comparison_group))
        self.wait(1)

        # 3. Fade out discarded elements (middle, right element, high pointer)
        self.play(
            FadeOut(middle_text),
            FadeOut(right_text),
            FadeOut(high_pointer),
            run_time=1.5 # Slightly longer fade for clarity
        )
        self.wait(1.5) # Keep the final state visible for a moment
