```python
from manim import *
import numpy as np

class Frame(Scene):
    def construct(self):
        # 1. Single Page Turning Simulation
        page = Rectangle(
            height=5.0,
            width=7.0,
            color="#CCCCCC", # Light gray border
            fill_color="#F0F0F0", # Off-white fill
            fill_opacity=1.0,
            stroke_width=2
        )
        # Add faint lines to simulate text lines on a page
        lines = VGroup(*[
            Line(
                page.get_corner(DL) + RIGHT * 0.5 + UP * (0.5 + i * 0.3),
                page.get_corner(DR) + LEFT * 0.5 + UP * (0.5 + i * 0.3),
                stroke_color="#AAAAAA", # Darker gray for lines
                stroke_width=1
            )
            for i in range(12) # Number of lines
        ])
        book_page = VGroup(page, lines).move_to(np.array([0,0,0])) # Use ORIGIN equivalent

        # Animate page appearing and flipping
        self.play(FadeIn(book_page), run_time=0.7)
        self.wait(0.3)
        # Pivot point for rotation (spine of the book)
        pivot_point = book_page.get_center() + LEFT * page.width / 2

        self.play(
            Rotate(book_page, angle=PI / 4, axis=UP, about_point=pivot_point),
            run_time=0.4
        )
        # Slight pause during flip
        self.wait(0.1)
        self.play(
            Rotate(book_page, angle=-PI / 4, axis=UP, about_point=pivot_point),
            run_time=0.4
        )
        self.wait(0.5)
        # Fade out the page
        self.play(FadeOut(book_page), run_time=0.5)
        self.wait(0.2) # Brief pause

        # 2. Horizontal List Appearance
        texts = ['two', 'five', 'eight', 'twelve', 'sixteen', 'twenty three', 'thirty eight', 'fifty six']
        text_mobjects = VGroup(*[
            Text(t, font_size=36, color="#EEEEEE") for t in texts # Slightly off-white text
        ])
        text_mobjects.arrange(RIGHT, buff=0.4) # Arrange horizontally with buffer

        # Create brackets as Text objects
        left_bracket = Text("[", font_size=48, color="#EEEEEE")
        right_bracket = Text("]", font_size=48, color="#EEEEEE")

        # Group text and brackets
        list_group = VGroup(left_bracket, text_mobjects, right_bracket)
        # Arrange the group including brackets
        list_group.arrange(RIGHT, buff=0.2)
        # Center the entire list group on the screen
        list_group.move_to(np.array([0,0,0])) # Use ORIGIN equivalent

        # Animate the list appearing
        self.play(FadeIn(list_group))
        # Keep the list on screen at the end
        self.wait(3)
```
