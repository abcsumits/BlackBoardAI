from manim import *

class Frame(Scene):
    def construct(self):
        # 1. Define elements
        num_23 = Text('twenty three', color='#FFFFFF', font_size=36)
        num_38 = Text('thirty eight', color='#FFFFFF', font_size=36)
        num_56 = Text('fifty six', color='#FFFFFF', font_size=36)

        # Group the numbers initially (can be placed anywhere before centering)
        number_group = VGroup(num_23, num_38, num_56).arrange(RIGHT, buff=1.0)
        number_group.move_to(ORIGIN + DOWN * 1.0) # Start slightly lower

        context_text = Text(
            "now repeat the steps find the middle of the remaining smaller list.",
            font_size=28,
            color='#CCCCCC'
        ).to_edge(UP, buff=0.5)

        # Display the initial list and context text
        self.play(Write(number_group), Write(context_text), run_time=1.5)
        self.wait(0.5)

        # 2. Center the group
        self.play(number_group.animate.move_to(ORIGIN), run_time=1.0)
        self.wait(0.5)

        # 3. Highlight the middle element ('thirty eight')
        middle_element = number_group[1] # Access the middle Text object
        highlight_dot = Dot(
            point=middle_element.get_bottom() + DOWN * 0.4, # Position dot below the text
            radius=0.15,
            color='#FFFF00' # Yellow
        )

        # Animate the dot appearing
        self.play(GrowFromCenter(highlight_dot), run_time=0.8)
        self.wait(2) # Hold the final visual state
