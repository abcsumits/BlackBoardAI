```python
from manim import *

class Frame(Scene):
    def construct(self):
        # Configuration
        text_color = "#FFFFFF" # White
        highlight_color = "#FFFF00" # Yellow
        equals_color = "#FFFFFF" # White
        checkmark_color = "#00FF00" # Green
        font_size = 60

        # 1. Display the final remaining element
        final_element_text = Text("twenty three", color=text_color, font_size=font_size)
        final_element_text.move_to(UP * 1.5)
        self.play(Write(final_element_text))
        self.wait(0.5)

        # 2. Highlight the element with a Dot
        highlight_dot = Dot(point=final_element_text.get_bottom() + DOWN * 0.3, color=highlight_color, radius=0.1)
        self.play(GrowFromCenter(highlight_dot))
        self.play(Indicate(final_element_text, color=highlight_color)) # Optional: briefly indicate the text
        self.wait(0.5)

        # 3. Create and show the equation parts
        element_copy = Text("twenty three", color=text_color, font_size=font_size)
        equals_sign = Text("=", color=equals_color, font_size=font_size)
        target_text = Text("twenty three", color=text_color, font_size=font_size)

        # Arrange equation horizontally and position it below the original text
        equation_group = VGroup(element_copy, equals_sign, target_text).arrange(RIGHT, buff=0.4)
        equation_group.next_to(final_element_text, DOWN, buff=1.0)

        # Animate the equation appearance
        self.play(
            Write(element_copy),
            Write(equals_sign),
            Write(target_text)
         )
        self.wait(0.5)

        # 4. Show the green checkmark
        checkmark = Checkmark(color=checkmark_color, stroke_width=10).scale(1.8)
        # Position checkmark above the equation, slightly higher than the original text was
        checkmark.next_to(equation_group, UP, buff=1.2)

        # Animate checkmark and fade out highlight dot and original text
        self.play(
            FadeOut(highlight_dot),
            FadeOut(final_element_text),
            Write(checkmark)
            )

        # Keep final elements (equation and checkmark) on screen
        self.wait(3)
```
