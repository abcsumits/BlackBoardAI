from manim import *

class Frame(Scene):
    def construct(self):
        # 1. Create Text elements
        text_twenty_three = Text("twenty three", color="#FFFFFF").scale(0.8)
        text_sixteen = Text("sixteen", color="#FFFFFF").scale(0.8)

        # 2. Create initial unequal sign (using MathTex for better look)
        # Position it roughly between where the texts will be
        unequal_sign = MathTex("!=", color="#FFFF00").scale(1.5)
        unequal_sign.move_to(ORIGIN)

        # Position texts relative to the sign initially
        text_twenty_three.next_to(unequal_sign, LEFT, buff=0.5)
        text_sixteen.next_to(unequal_sign, RIGHT, buff=0.5)

        # Group initial elements for simultaneous animation
        initial_elements = VGroup(text_twenty_three, unequal_sign, text_sixteen)

        # 3. Show initial state (texts and != sign)
        self.play(Write(text_twenty_three), Write(text_sixteen), Write(unequal_sign))
        self.wait(0.5)

        # 4. Fade out the unequal sign
        self.play(FadeOut(unequal_sign))
        # Keep track of the sign's position for the replacement
        sign_position = unequal_sign.get_center()

        # 5. Create GreaterThan sign using Line objects
        tip_point = sign_position + RIGHT * 0.3
        top_point = sign_position + LEFT * 0.3 + UP * 0.4
        bottom_point = sign_position + LEFT * 0.3 + DOWN * 0.4

        line1 = Line(start=top_point, end=tip_point, stroke_width=8, color="#FFFF00")
        line2 = Line(start=bottom_point, end=tip_point, stroke_width=8, color="#FFFF00")
        greater_than_sign = VGroup(line1, line2)

        # 6. Position the greater_than_sign where the unequal sign was
        # Ensure text elements are positioned correctly relative to the new sign
        text_twenty_three.generate_target()
        text_sixteen.generate_target()
        text_twenty_three.target.next_to(greater_than_sign, LEFT, buff=0.5)
        text_sixteen.target.next_to(greater_than_sign, RIGHT, buff=0.5)

        # 7. Animate the appearance of '>' and the repositioning of text
        self.play(
            Create(greater_than_sign),
            MoveToTarget(text_twenty_three),
            MoveToTarget(text_sixteen)
        )
        self.wait(0.5)

        # 8. Pulse 'twenty three'
        self.play(Indicate(text_twenty_three, scale_factor=1.2, color="#FFFFFF")) # Use Indicate for pulse effect
        self.wait(1.5) # Hold the final frame longer
