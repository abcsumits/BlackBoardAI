from manim import *

class Frame(Scene):
    def construct(self):
        # Define vertices for a simplified thumbs-up shape
        # Points arranged to form the outline
        thumb_points = [
            [-1.0, -1.0, 0],  # Bottom left base
            [1.0, -1.0, 0],   # Bottom right base
            [1.0, 0.5, 0],   # Top right base knuckle area
            [0.6, 0.5, 0],   # Inner thumb base start
            [0.6, 1.8, 0],   # Inner thumb side
            [0.2, 2.3, 0],   # Thumb tip top point
            [-0.2, 1.8, 0],  # Outer thumb side
            [-0.2, 0.5, 0],  # Outer thumb base end
            [-1.0, 0.5, 0]   # Top left base
        ]

        # Create the Polygon object for the thumbs-up
        thumbs_up = Polygon(*thumb_points, color="#008000", fill_opacity=1.0)
        thumbs_up.scale(1.3) # Scale it to be large

        # Create the Star object
        star = Star(n=5, outer_radius=1.0, inner_radius=0.5, color="#FFFF00", fill_opacity=1.0)
        star.scale(1.3) # Scale it to be large

        # Position the shapes to avoid overlap
        # Shift thumbs-up left and star right
        thumbs_up.shift(LEFT * 2.5)
        star.shift(RIGHT * 2.5)

        # Animate the shapes appearing with FadeIn
        self.play(FadeIn(thumbs_up), FadeIn(star))

        # Wait for a moment to show the final state
        self.wait(2)
