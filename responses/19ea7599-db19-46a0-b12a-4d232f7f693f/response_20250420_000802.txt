```python
from manim import *

class Frame(Scene):
    def construct(self):
        # Define hex colors
        orange_color = '#FFA500'
        gray_color = '#808080'

        # Create Text objects (logos)
        forge_logo = Text('Forge', color=orange_color).scale(1.5)
        fabric_logo = Text('Fabric', color=gray_color).scale(1.5)

        # Position the logos side-by-side without overlap
        forge_logo.shift(LEFT * 2.5)
        fabric_logo.shift(RIGHT * 2.5)

        # Display the logos
        self.play(Write(forge_logo), Write(fabric_logo))
        self.wait(0.5)

        # Define the pulsing animation
        # A gentle scale up and down using ScaleInPlace and there_and_back rate function
        pulse_scale = 1.1
        pulse_runtime = 1.5

        # Create the pulsing animations for both logos
        forge_pulse = ScaleInPlace(forge_logo, pulse_scale, rate_func=there_and_back, run_time=pulse_runtime)
        fabric_pulse = ScaleInPlace(fabric_logo, pulse_scale, rate_func=there_and_back, run_time=pulse_runtime)

        # Group the animations to play them simultaneously
        pulse_animation_group = AnimationGroup(forge_pulse, fabric_pulse)

        # Play the pulsing animation a few times
        self.play(pulse_animation_group)
        self.play(pulse_animation_group)
        self.play(pulse_animation_group)


        # Keep the logos on screen at the end
        self.wait(1)

```
