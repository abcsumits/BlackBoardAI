```python
from manim import *

class Frame(Scene):
    def construct(self):
        # Define colors using hex codes
        text_color_green = "#84C3BE" # Manim's GREEN
        text_color_white = "#FFFFFF" # White

        # Define the title text (optional context)
        title = Text("What is MCP Server?", color=text_color_white, font_size=36)
        title.to_edge(UP, buff=1.0)

        # Define the checklist item text
        checklist_item = Text(
            'MCP Server ≈ Modded Server (Forge/Fabric)',
            color=text_color_green,
            font_size=32
        )
        checklist_item.next_to(title, DOWN, buff=0.7)

        # Define the checkmark
        checkmark = Checkmark(color=text_color_green)
        checkmark.scale(0.8) # Scale for appropriate size
        checkmark.next_to(checklist_item, LEFT, buff=0.3) # Position left of the text

        # Animations
        self.play(Write(title), run_time=1.0)
        self.play(Write(checklist_item), run_time=1.5)
        self.play(Create(checkmark), run_time=1.0)

        # Keep the final scene visible for a moment
        self.wait(2)
```
