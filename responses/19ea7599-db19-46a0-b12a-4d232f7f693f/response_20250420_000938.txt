```python
from manim import *
import numpy as np

class Frame(Scene):
    def construct(self):
        # Define colors using hex codes
        phone_border_color = '#000000'
        phone_fill_color = '#F5F5F5'
        ground_color = '#A0522D' # SaddleBrown
        grass_color = '#228B22'  # ForestGreen
        block_stroke_color = '#000000'
        text_color = '#0000FF'   # Blue

        # Create the phone screen shape
        phone_height = 4.5
        phone_width = phone_height / 1.9
        phone_screen = Rectangle(
            width=phone_width,
            height=phone_height,
            color=phone_border_color,
            fill_color=phone_fill_color,
            fill_opacity=1.0,
            stroke_width=2
        )
        # Position the phone screen slightly above the center
        phone_screen.move_to(np.array([0, 0.5, 0]))

        # Calculate dimensions for the block world inside the phone
        pc = phone_screen.get_center()
        inner_padding = 0.2 # Increase padding slightly
        drawable_height = phone_height - 2 * inner_padding
        drawable_width = phone_width - 2 * inner_padding

        # Determine block size based on width
        num_blocks_wide = 5
        block_size = drawable_width / num_blocks_wide

        # Ensure blocks aren't too tall for the space
        max_block_height = drawable_height / 2 # Allow for 2 layers: ground + grass
        if block_size > max_block_height:
            block_size = max_block_height

        # Calculate Y position for the ground layer (bottom layer inside phone)
        ground_y = pc[1] - drawable_height / 2 + block_size / 2

        # Create ground blocks
        ground_blocks = VGroup()
        for i in range(num_blocks_wide):
            # Calculate X position for each block
            x_pos = pc[0] - drawable_width / 2 + block_size / 2 + i * block_size
            ground_block = Square(
                side_length=block_size,
                color=ground_color,
                fill_opacity=1.0,
                stroke_width=1.5, # Slightly thicker stroke
                stroke_color=block_stroke_color
            )
            ground_block.move_to(np.array([x_pos, ground_y, 0]))
            ground_blocks.add(ground_block)

        # Create grass blocks on top of ground blocks
        grass_blocks = VGroup()
        grass_y = ground_y + block_size # Y position for the grass layer
        # Check if grass layer fits within the drawable height
        if grass_y + block_size / 2 <= pc[1] + drawable_height / 2:
             for ground_block in ground_blocks:
                grass_block = Square(
                    side_length=block_size,
                    color=grass_color,
                    fill_opacity=1.0,
                    stroke_width=1.5, # Slightly thicker stroke
                    stroke_color=block_stroke_color
                )
                # Position grass block directly above its corresponding ground block
                grass_block.move_to(ground_block.get_center() + np.array([0, block_size, 0]))
                grass_blocks.add(grass_block)

        # Group the block world elements
        block_world = VGroup(ground_blocks, grass_blocks)
        # Ensure block world is drawn above the phone screen background but below border? No, inside. Z-index not needed.

        # Create the text object
        mcpe_text = Text('MCPE / Bedrock', color=text_color, font_size=36)
        # Position the text below the phone screen
        mcpe_text.next_to(phone_screen, DOWN, buff=0.4)

        # --- Animations ---

        # Animate the phone screen appearing
        self.play(Create(phone_screen), run_time=1.0)
        self.wait(0.2)

        # Animate the block world appearing (ground first, then grass)
        self.play(Create(ground_blocks), run_time=0.8)
        if len(grass_blocks) > 0: # Only animate grass if it was created
             self.play(Create(grass_blocks), run_time=0.8)
        self.wait(0.5)

        # Animate the text fading in
        self.play(FadeIn(mcpe_text), run_time=1.0)

        # Hold the final scene for a couple of seconds
        self.wait(2)
```
