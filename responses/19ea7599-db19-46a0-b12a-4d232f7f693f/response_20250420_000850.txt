from manim import *

class Frame(Scene):
    def construct(self):
        # Define Text elements with specified hex colors
        mcp_text = Text('MCP', color='#00FFFF') # CYAN
        mods_text = Text('Mods', color='#00FF00') # GREEN
        experience_text = Text('Modded Server Experience', color='#FFFF00') # YELLOW

        # Position elements horizontally to avoid overlap
        # Ensure enough space between them for clarity and arrows
        mcp_text.shift(LEFT * 4.5)
        # mods_text stays near the center
        experience_text.shift(RIGHT * 5.5)

        # Adjust vertical alignment if needed, but horizontal is simpler here
        # For example, aligning baseline: group = VGroup(mcp_text, mods_text, experience_text).arrange(RIGHT, buff=1.5)

        # Create Arrows connecting the text elements
        # Use buff to add a small gap between text and arrow tip/tail
        arrow1 = Arrow(
            mcp_text.get_right(),
            mods_text.get_left(),
            buff=0.2,
            color=WHITE # Default arrow color
        )
        arrow2 = Arrow(
            mods_text.get_right(),
            experience_text.get_left(),
            buff=0.2,
            color=WHITE # Default arrow color
        )

        # Animate the creation of elements sequentially
        self.play(Write(mcp_text))
        self.wait(0.3)
        self.play(Write(mods_text))
        self.play(GrowArrow(arrow1))
        self.wait(0.3)
        self.play(Write(experience_text))
        self.play(GrowArrow(arrow2))

        # Hold the final scene for a moment
        self.wait(1)
