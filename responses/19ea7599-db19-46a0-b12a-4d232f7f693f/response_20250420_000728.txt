from manim import *

class Frame(Scene):
    def construct(self):
        # Define initial simple shapes with hex colors
        circle = Circle(color="#3498DB", fill_opacity=0.8).scale(0.8)
        square = Square(color="#2ECC71", fill_opacity=0.8).scale(0.8)
        triangle = Triangle(color="#E74C3C", fill_opacity=0.8).scale(0.8)

        # Group and position initial shapes horizontally
        initial_shapes = VGroup(circle, square, triangle).arrange(RIGHT, buff=1.5)
        self.play(LaggedStart(*[Create(shape) for shape in initial_shapes], lag_ratio=0.5))
        self.wait(1)

        # Define target polygon shapes with hex colors
        # Ensure target shapes align with initial shapes' positions
        target_star = Star(n=7, outer_radius=1.0, inner_radius=0.5, color="#85C1E9", fill_opacity=0.8).move_to(circle.get_center())
        target_hexagon = RegularPolygon(n=6, color="#7DCEA0", fill_opacity=0.8).scale(1.0).move_to(square.get_center())
        # More complex polygon for triangle's transformation
        target_complex_poly = Polygon(
            [-1, 1, 0], [1, 1, 0], [1.5, 0, 0], [1, -1, 0],
            [-1, -1, 0], [-1.5, 0, 0],
            color="#F1948A", fill_opacity=0.8
        ).scale(0.7).move_to(triangle.get_center())


        # Animate the transformation
        self.play(
            Transform(circle, target_star),
            Transform(square, target_hexagon),
            Transform(triangle, target_complex_poly),
            run_time=2
        )
        self.wait(2)
