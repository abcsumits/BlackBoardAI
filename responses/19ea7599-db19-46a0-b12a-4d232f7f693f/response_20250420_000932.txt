from manim import *

class Frame(Scene):
    def construct(self):
        # Define colors using hex codes
        server_color = "#808080" # Grey
        phone_color = "#A9A9A9" # DarkGrey
        console_color = "#D3D3D3" # LightGrey
        line_color = "#FFFFFF" # White
        dot_color_on = "#00FF00" # Green
        dot_color_off = "#006400" # DarkGreen
        text_color = "#FFFFFF" # White

        # Title Text (Explaining the context implicitly)
        # title = Text("What is MCP Server?", color=text_color).scale(0.8).to_edge(UP)
        # self.play(Write(title))
        # self.wait(0.5)

        # Server Icon Creation
        server_body = Rectangle(width=1.5, height=2.5, color=server_color, fill_opacity=1.0)
        server_dot = Dot(color=dot_color_on, radius=0.1).move_to(server_body.get_center() + DOWN * 0.8 + LEFT * 0.5)
        # Group and Position server icon slightly left of center
        server = VGroup(server_body, server_dot).shift(LEFT * 3.5)

        # Phone Icon Creation (Using Rectangle as a simple polygon)
        phone_shape = Rectangle(
            width=1.0, height=1.8, color=phone_color, fill_opacity=1.0
        ).shift(RIGHT * 3 + UP * 1.5) # Position right and up

        # Console Icon Creation (Using Polygon for a distinct shape)
        # Vertices define a shape slightly wider at the top, resembling a console base
        console_shape = Polygon(
             [-1.2, 0.4, 0], [1.2, 0.4, 0], [1.0, -0.4, 0], [-1.0, -0.4, 0],
             color=console_color, fill_opacity=1.0
        ).scale(0.8).shift(RIGHT * 3 + DOWN * 1.5) # Position right and down


        # Connection Lines Creation
        line_to_phone = Line(
            server.get_right(), phone_shape.get_left(), # Connect server right edge to phone left edge
            color=line_color, stroke_width=3
        )
        line_to_console = Line(
            server.get_right(), console_shape.get_left(), # Connect server right edge to console left edge
            color=line_color, stroke_width=3
        )

        # Animation Sequence
        self.play(Create(server_body), run_time=1.0) # Animate server body creation
        self.play(FadeIn(server_dot), run_time=0.5) # Animate dot appearance

        # Introduce client devices (phone and console)
        self.play(Create(phone_shape), Create(console_shape), run_time=1.5) # Animate creation simultaneously
        # Animate connection lines appearing
        self.play(Create(line_to_phone), Create(line_to_console), run_time=1.5)

        # Blinking Animation for the Server Dot
        # Loop executes a specific number of times, not infinitely
        for _ in range(4): # Make the dot blink 4 times
            # Animate dot color change to 'off' state
            self.play(server_dot.animate.set_color(dot_color_off), run_time=0.25)
            # Animate dot color change back to 'on' state
            self.play(server_dot.animate.set_color(dot_color_on), run_time=0.25)

        # Keep the final scene visible for a moment before ending
        self.wait(2)
