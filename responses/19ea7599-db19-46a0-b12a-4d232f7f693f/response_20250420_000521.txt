from manim import *
import numpy as np

class Frame(Scene):
    def construct(self):
        # 1. Create Axes and Graph
        axes = Axes(
            x_range=[0, 10, 1],
            y_range=[-2, 2, 1], # Adjusted y-range for visibility
            x_length=7,
            y_length=4,
            axis_config={"color": "#CCCCCC"}, # Light grey axes
            tips=False, # No arrow tips on axes
        ).move_to(np.array([0,0,0])) # Center the axes using np.array

        # Define a relatively smooth function for the initial graph
        def func(x):
            return np.sin(x * 0.8) * 1.5 # Smooth sine wave

        graph = axes.plot(func, color="#4CAF50", stroke_width=3) # Green color, slightly thicker line

        # Animate the creation of axes and graph
        self.play(Create(axes), Create(graph), run_time=1.5)
        self.wait(0.5)

        # 2. Wobble Animation
        wobble_duration = 2.5 # Duration of the wobble
        noise_amplitude = 1.0 # Controls the intensity of the wobble per second

        # Define the updater function for the wobble effect
        def update_graph_wobble(mob, dt):
            # Get the current points of the graph
            current_points = mob.get_points()
            if len(current_points) > 0:
                # Generate random vertical displacement proportional to dt
                # Use randn for normal distribution, scaled by amplitude and time delta
                perturbation = (np.random.randn(len(current_points)) * noise_amplitude * dt)

                # Apply the perturbation only to the y-coordinate (index 1)
                current_points[:, 1] += perturbation

                # Update the graph's points
                mob.set_points(current_points)

        # Add the updater to the graph
        graph.add_updater(update_graph_wobble)
        # Wait for the duration of the wobble animation
        self.wait(wobble_duration)
        # Remove the updater to stop the wobble
        graph.remove_updater(update_graph_wobble)
        self.wait(0.2) # Short pause after wobbling stops

        # 3. Morph into Tangled Ball
        tangled_lines_group = VGroup() # Use VGroup to hold the lines
        num_lines = 40 # Number of lines in the tangled ball
        tangle_radius = 1.6 # Radius of the area for the tangled ball

        # Create multiple Line objects for the tangled ball
        for _ in range(num_lines):
            # Generate random start and end points within a spherical region
            center_offset = np.random.randn(3) * tangle_radius * 0.4 # Offset from center
            start_point = center_offset + np.random.randn(3) * 0.5 # Add variation
            end_point = center_offset + np.random.randn(3) * 0.5   # Add variation

            line = Line(
                start_point,
                end_point,
                color="#E0E0E0", # Off-white color for lines
                stroke_width=1.5 # Thinner lines for a tangled look
             )
            tangled_lines_group.add(line)

        # Position the tangled ball group at the center
        tangled_lines_group.move_to(np.array([0,0,0]))
        # Start the ball scaled down and invisible for the transform effect
        tangled_lines_group.scale(0.01)
        tangled_lines_group.set_opacity(0)

        # Animate the transformation:
        # Fade out the axes downwards
        # Replace the graph with the tangled ball, scaling it up and fading it in
        self.play(
            FadeOut(axes, shift=np.array([0,-1,0]), scale=0.8, run_time=2.0), # Fade out axes with shift and scale
            ReplacementTransform(
                graph, # Transform the graph
                tangled_lines_group.animate.scale(100).set_opacity(1), # Into the ball (scale factor matches initial scale)
                run_time=2.5 # Duration of the morph
            )
        )

        # Hold the final tangled ball state
        self.wait(1.5)
