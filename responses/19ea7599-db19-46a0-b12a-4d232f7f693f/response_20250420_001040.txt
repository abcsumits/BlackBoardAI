```python
from manim import *

class Frame(Scene):
    def construct(self):
        # Define text and color
        text_color = '#00FF00'  # GREEN hex code
        item_text_str = 'MCP Server ≈ Modded Server (Forge/Fabric)'
        item_text = Text(item_text_str, color=text_color, font_size=36)

        # Create checkmark
        checkmark = Checkmark(color=text_color)
        checkmark.scale(0.8) # Scale checkmark to be slightly smaller than text height

        # Group text and position checkmark relative to it
        # Arrange checkmark to the left of the text
        group = VGroup(checkmark, item_text)
        group.arrange(LEFT, buff=0.4) # Add some buffer between checkmark and text

        # Position the entire group in the center of the frame
        group.move_to(ORIGIN)

        # Animate the text appearance first
        self.play(Write(item_text), run_time=1.5)
        self.wait(0.5) # Brief pause before checkmark appears

        # Animate the checkmark creation
        # Position checkmark before creation animation starts
        checkmark.move_to(group[0].get_center())
        self.play(Create(checkmark), run_time=1)

        # Hold the final visual on screen
        self.wait(2)
```
