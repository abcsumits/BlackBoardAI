```python
from manim import *

class Frame(Scene):
    def construct(self):
        # Define the main text centered on the screen
        main_text = Text('MCP Server = ?', color='#FFFFFF').scale(1.2)
        main_text.move_to([0, 0, 0]) # Position at the origin (center)

        # Define the surrounding acronyms with positions relative to the center
        acronym1 = Text('MCPE?', color='#FFFFFF').scale(0.8)
        acronym1.move_to([-3, 1.5, 0]) # Position Top-Left quadrant

        acronym2 = Text('XYZ?', color='#FFFFFF').scale(0.8)
        acronym2.move_to([3, -1.5, 0]) # Position Bottom-Right quadrant

        acronym3 = Text('MCN?', color='#FFFFFF').scale(0.8) # Another potential acronym
        acronym3.move_to([3, 1.5, 0])  # Position Top-Right quadrant

        # Animation sequence starts here
        # 1. Write the main central text
        self.play(Write(main_text), run_time=1.5)
        self.wait(0.5)

        # 2. Fade in the first acronym ('MCPE?')
        self.play(FadeIn(acronym1, shift=UP*0.5), run_time=0.7)
        self.wait(1.2) # Pause to let viewer read

        # 3. Fade in the second acronym ('XYZ?')
        self.play(FadeIn(acronym2, shift=DOWN*0.5), run_time=0.7)
        self.wait(0.7) # Shorter pause

        # 4. Fade out the first acronym and fade in the third ('MCN?') simultaneously
        self.play(
            FadeOut(acronym1, shift=UP*0.5),
            FadeIn(acronym3, shift=UP*0.5),
            run_time=1.0 # Slightly longer runtime for combined animation
        )
        self.wait(1.2) # Pause again

        # 5. Fade out the second acronym ('XYZ?')
        self.play(FadeOut(acronym2, shift=DOWN*0.5), run_time=0.7)
        self.wait(0.5)

        # 6. Fade out the third acronym ('MCN?')
        self.play(FadeOut(acronym3, shift=UP*0.5), run_time=0.7)

        # End scene with the main text still visible
        self.wait(1.5)
```
