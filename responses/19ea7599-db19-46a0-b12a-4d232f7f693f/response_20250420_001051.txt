from manim import *

class Frame(Scene):
    def construct(self):
        # Define colors
        forge_color = "#FFA500"  # Orange
        fabric_color = "#808080" # Gray
        text_color = "#FFFFFF"   # White

        # Create Forge Icon representation (Stylized 'F')
        forge_icon = Text("F", color=forge_color, weight=BOLD).scale(3.5)

        # Create Fabric Icon representation (Stylized 'F')
        fabric_icon = Text("F", color=fabric_color, weight=BOLD).scale(3.5)

        # Create Text
        look_text = Text("Look for these!", color=text_color).scale(0.9)

        # Position elements
        forge_icon.move_to(LEFT * 3 + UP * 0.5)
        fabric_icon.move_to(RIGHT * 3 + UP * 0.5)
        icons_group = VGroup(forge_icon, fabric_icon)
        look_text.next_to(icons_group, DOWN, buff=0.6)

        # Center the group vertically slightly if needed, but individual positioning should keep it centered
        final_group = VGroup(icons_group, look_text)
        # final_group.center() # Recenter the whole group if needed after relative positioning

        # Animation
        self.play(
            Create(forge_icon),
            Create(fabric_icon),
            run_time=1.5
        )
        self.play(
            Write(look_text),
            run_time=1.5
        )

        # Keep the final frame visible
        self.wait(2)
