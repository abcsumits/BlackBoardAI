from manim import *
import numpy as np
import random

class Frame(Scene):
    def construct(self):
        # Set background color
        self.camera.background_color = "#1a1a1a" # Dark gray background

        # 1. Create Axes
        axes = Axes(
            x_range=[-5, 5, 1],
            y_range=[-3, 3, 1],
            axis_config={"color": "#cccccc", "include_tip": False}, # Light gray axes, no tips
            x_length=10,
            y_length=6,
        ).add_coordinates()
        axes.move_to(np.array([0, 0, 0])) # Center axes explicitly

        # 2. Create Initial Graph
        def func(x):
            # A slightly complex but smooth function for the graph
            return 0.9 * np.sin(x) + 0.5 * np.sin(2.2 * x) + 0.3 * np.cos(3.5 * x)

        graph = axes.plot(
            func,
            x_range=[-4.5, 4.5],
            color="#4ec9b0", # Teal color
            use_smoothing=True, # Makes the curve smoother
            stroke_width=3.5
        )

        # Animate creation of axes and graph
        self.play(Create(axes), Create(graph), run_time=1.5)
        self.wait(0.5)

        # 3. Wobble Animation using Wiggle
        # Wiggle provides a simple and efficient way to show instability
        self.play(
            Wiggle(
                graph,
                scale_value=1.4,        # How much it scales during wiggle
                rotation_angle=0.015 * TAU, # How much it rotates
                n_wiggles=10,           # Number of oscillations
                run_time=2.5            # Duration of the wobble
            )
        )
        self.wait(0.5)

        # 4. Morph into a Tangle
        center_point = axes.get_origin() # Get the origin of the axes system (0,0,0)
        tangle_radius = 1.8 # Controls the spread of the final tangled ball

        # Define a function that maps each point of the graph to a new random position
        # This function creates the tangled effect by moving points towards the center randomly
        def tangle_point_map(point):
            # Calculate a random direction in the XY plane (z=0)
            random_direction = normalize(np.random.randn(3) * np.array([1, 1, 0]))
            # Assign a random distance from the center within the tangle_radius
            random_radius = random.uniform(0.2, tangle_radius)
            # Calculate the new target point relative to the center_point
            target_point = center_point + random_direction * random_radius
            return target_point

        # Animate the transformation using ApplyPointwiseFunction
        # This applies the tangle_point_map function to every point defining the graph
        self.play(
            ApplyPointwiseFunction(tangle_point_map, graph, run_time=2.5), # Morph the graph
            FadeOut(axes, run_time=1.5) # Fade out the axes simultaneously
        )
        # The 'graph' Mobject now visually represents the tangled ball of lines

        # Ensure the final tangled shape remains on screen
        self.wait(1.5)
