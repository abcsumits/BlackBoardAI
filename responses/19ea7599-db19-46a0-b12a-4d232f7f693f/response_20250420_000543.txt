from manim import *

class Frame(Scene):
    def construct(self):
        # Define colors
        red_color = '#FF0000'

        # Create the text object
        term_text = Text('MCP Server != Official Term', color=red_color)
        term_text.scale(0.9) # Scale slightly to ensure it fits well

        # Position the text (implicitly centered)
        # self.play(Write(term_text)) # Initial write if needed, but prompt asks for FadeIn

        # Fade in the text
        self.play(FadeIn(term_text))
        self.wait(0.5) # Brief pause after text appears

        # Calculate positions for the 'X' based on the text's bounding box
        padding = 0.2 # Add a little padding around the text for the X
        top_left = term_text.get_corner(UL) + UP * padding + LEFT * padding
        top_right = term_text.get_corner(UR) + UP * padding + RIGHT * padding
        bottom_left = term_text.get_corner(DL) + DOWN * padding + LEFT * padding
        bottom_right = term_text.get_corner(DR) + DOWN * padding + RIGHT * padding

        # Create the two lines for the 'X' mark
        line1 = Line(start=top_left, end=bottom_right, color=red_color, stroke_width=8)
        line2 = Line(start=top_right, end=bottom_left, color=red_color, stroke_width=8)

        # Animate the creation of the 'X' mark
        self.play(Create(line1), Create(line2))

        # Keep the final result on screen for a moment
        self.wait(1.5)
