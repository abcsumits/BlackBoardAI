from manim import *
import random
import numpy as np

class Frame(Scene):
    def construct(self):
        # Define constants for positioning, colors, and network properties
        network_center = ORIGIN
        network_area_radius = 2.5 # Define the radius of the area where dots can be placed
        num_dots = 15 # Target number of dots in the network
        dot_color = "#CCCCCC" # Light grey for dots
        line_color = "#AAAAAA" # Darker grey for lines
        cross_color = "#FF0000" # Red color for the 'X'
        cross_stroke_width = 15 # Thickness of the 'X' lines
        dot_radius = 0.08 # Size of the dots

        # Create dots within a circular area, attempting to avoid significant overlap
        dots = VGroup()
        dot_positions = []
        min_distance_sq = (dot_radius * 3)**2 # Minimum squared distance between dot centers
        attempts = 0
        max_attempts = num_dots * 15 # Limit attempts to prevent potential infinite loops

        while len(dot_positions) < num_dots and attempts < max_attempts:
            # Generate a random position within the defined circular area
            angle = random.uniform(0, 2 * PI)
            radius = random.uniform(0.1, network_area_radius) # Avoid placing dots exactly at the center
            pos = network_center + np.array([radius * np.cos(angle), radius * np.sin(angle), 0])
            
            # Check if the new position is too close to existing dots
            too_close = False
            for existing_pos in dot_positions:
                if np.sum((pos - existing_pos)**2) < min_distance_sq:
                    too_close = True
                    break
            
            # If the position is suitable, create and add the dot
            if not too_close:
                dot = Dot(point=pos, color=dot_color, radius=dot_radius)
                dots.add(dot)
                dot_positions.append(pos)
            attempts += 1
        
        # Update num_dots to the actual number of dots placed
        num_dots = len(dot_positions) 

        # Create lines connecting random pairs of dots, only if there are at least two dots
        lines = VGroup()
        if num_dots > 1:
            # Calculate the maximum possible number of unique lines
            max_possible_lines = num_dots * (num_dots - 1) // 2
            # Aim for a number of lines proportional to the number of dots for visual complexity
            num_lines_target = int(num_dots * 1.2) 
            # Ensure the number of lines does not exceed the maximum possible
            num_lines = min(num_lines_target, max_possible_lines) 
            
            # Generate all possible pairs of dot indices
            potential_pairs = [(i, j) for i in range(num_dots) for j in range(i + 1, num_dots)]
            # Shuffle the pairs to get random connections
            random.shuffle(potential_pairs)

            # Create lines for the selected number of pairs
            for i in range(min(num_lines, len(potential_pairs))): # Use min for safety
                 idx1, idx2 = potential_pairs[i]
                 line = Line(dot_positions[idx1], dot_positions[idx2], color=line_color, stroke_width=2)
                 lines.add(line)

        # Group the lines and dots into a single network object
        # Drawing lines first ensures dots appear on top of line endpoints
        network = VGroup(lines, dots) 
        
        # Proceed only if network elements were successfully created
        if len(network) > 0: 
             # Center the entire network diagram in the frame
             network.move_to(ORIGIN) 

             # Animation: Create the network diagram on screen
             self.play(Create(network), run_time=1.5)
             # Pause briefly to show the network
             self.wait(0.75) 

             # Create the large red 'X' to overlay the network
             # Determine the size of the X based on the network's dimensions
             network_width = network.width
             network_height = network.height
             # The cross should be slightly larger than the network's largest dimension
             cross_dimension = max(network_width, network_height) 
             # Add some padding and ensure a minimum size for visibility
             cross_dimension = max(cross_dimension * 1.1, 3.0) 

             # Create the two lines forming the 'X' using Manim direction vectors
             line1 = Line(LEFT * cross_dimension / 2 + UP * cross_dimension / 2,
                          RIGHT * cross_dimension / 2 + DOWN * cross_dimension / 2,
                          color=cross_color, stroke_width=cross_stroke_width)
             line2 = Line(LEFT * cross_dimension / 2 + DOWN * cross_dimension / 2,
                          RIGHT * cross_dimension / 2 + UP * cross_dimension / 2,
                          color=cross_color, stroke_width=cross_stroke_width)
             # Group the lines into a single 'X' object
             cross = VGroup(line1, line2)
             # Position the center of the 'X' over the center of the network
             cross.move_to(network.get_center()) 

             # Animation: Make the 'X' appear, growing from its center
             self.play(GrowFromCenter(cross), run_time=0.75)
             
             # Final pause to show the network overlaid with the 'X'
             self.wait(1.5)
        else:
             # Fallback message if no dots could be placed (e.g., due to extreme spacing constraints)
             fallback_text = Text("Network visualization unavailable", color="#FFFF00", font_size=24)
             self.play(Write(fallback_text))
             self.wait(2)
