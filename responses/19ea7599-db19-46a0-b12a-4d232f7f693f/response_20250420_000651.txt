```python
from manim import *

class Frame(Scene):
    def construct(self):
        # Define hex color for CYAN
        cyan_hex = "#00FFFF"

        # Create the first text object
        text1 = Text('MCP = Mod Coder Pack', color=cyan_hex)
        text1.scale(0.8) # Scale down slightly if needed
        text1.shift(UP * 0.5) # Position slightly above center

        # Create the second text object
        text2 = Text('/ Minecraft Coder Pack', color=cyan_hex)
        text2.scale(0.8) # Scale down slightly if needed
        text2.next_to(text1, DOWN, buff=0.3) # Position below the first text

        # Animate the first text appearing
        self.play(Write(text1), run_time=1.5)
        self.wait(0.5)

        # Animate the second text typing out
        self.play(AddTextLetterByLetter(text2), run_time=2.5)
        self.wait(2) # Keep the final frame visible
```
