```python
from manim import *

class Frame(Scene):
    def construct(self):
        # Define colors
        bubble_stroke_color = "#333333" # Dark grey
        bubble_fill_color = "#FFFFFF"   # White
        question_mark_color = "#000000" # Black
        dot_color = "#555555"           # Grey

        # Configuration
        bubble_radius = 1.5
        bubble_center_pos = UP * 1.0 # Initial position relative to ORIGIN
        dot_start_offset = DOWN * 0.4 + LEFT * 0.2 # Relative to bubble bottom
        dot_spacing = DOWN * 0.4 + LEFT * 0.2

        # Create main bubble
        bubble = Circle(
            radius=bubble_radius,
            stroke_color=bubble_stroke_color,
            stroke_width=6,
            fill_color=bubble_fill_color,
            fill_opacity=1.0
        )
        bubble.move_to(bubble_center_pos)

        # Create question mark
        question_mark = Text("?", font_size=144, color=question_mark_color, weight=BOLD)
        # We will position the question mark after the bubble is positioned globally

        # Create trailing dots
        # Initial positions based on bubble's initial position
        dot1_pos = bubble.get_bottom() + dot_start_offset
        dot1 = Dot(point=dot1_pos, radius=0.15, color=dot_color)

        dot2_pos = dot1.get_center() + dot_spacing
        dot2 = Dot(point=dot2_pos, radius=0.10, color=dot_color)

        dot3_pos = dot2.get_center() + dot_spacing
        dot3 = Dot(point=dot3_pos, radius=0.05, color=dot_color)

        dots = VGroup(dot3, dot2, dot1) # Group smallest to largest for animation order

        # Group bubble and dots for positioning
        bubble_and_dots = VGroup(bubble, dots)

        # Position the entire visual slightly above the screen center
        bubble_and_dots.move_to(ORIGIN + UP*0.5)

        # Now position the question mark inside the correctly positioned bubble
        question_mark.move_to(bubble.get_center())

        # Animation sequence
        # Animate dots appearing sequentially, from smallest to largest
        self.play(
            Create(dot3),
            run_time=0.3
        )
        self.play(
            Create(dot2),
            run_time=0.3
        )
        self.play(
            Create(dot1),
            run_time=0.3
        )
        # Animate bubble appearing
        self.play(
           GrowFromCenter(bubble),
            run_time=0.8
        )
        # Animate question mark appearing inside the bubble
        self.play(
            Write(question_mark),
            run_time=0.8
        )

        # Hold the final scene for a couple of seconds
        self.wait(2)
```
