```python
from manim import *
import numpy as np

class Frame(Scene):
    def construct(self):
        # Configuration
        text_color_hex = "#FF0000" # Red
        checkmark_color_hex = "#34A853" # Green
        text_scale = 0.8
        checkmark_scale = 0.8

        # Create the text item
        checklist_text = Text('MCP Server: Not Official', color=text_color_hex)
        checklist_text.scale(text_scale)

        # Position the text item
        # Slightly left to make space for the checkmark, and slightly up
        checklist_text.shift(UP * 0.5 + LEFT * 2.5)

        # Display the text item
        self.play(Write(checklist_text), run_time=1.0)
        self.wait(0.5)

        # Create the checkmark components
        checkmark_part1 = Line(
            start=np.array([-0.25, -0.25, 0]),
            end=np.array([0, 0, 0]),
            color=checkmark_color_hex,
            stroke_width=8 # Make checkmark thicker
        )
        checkmark_part2 = Line(
            start=np.array([0, 0, 0]),
            end=np.array([0.5, -0.5, 0]),
            color=checkmark_color_hex,
            stroke_width=8 # Make checkmark thicker
        )

        # Group the checkmark parts
        checkmark = VGroup(checkmark_part1, checkmark_part2)
        checkmark.scale(checkmark_scale)

        # Position the checkmark next to the text
        checkmark.next_to(checklist_text, RIGHT, buff=0.4)
        # Adjust vertical alignment to roughly match text center
        checkmark.shift(DOWN * 0.05)

        # Animate the checkmark drawing
        self.play(Create(checkmark_part1), run_time=0.3)
        self.play(Create(checkmark_part2), run_time=0.3)

        # Hold the final scene
        self.wait(2)
```
