```python
from manim import *

class Frame(Scene):
    def construct(self):
        # Define the text for the checklist item
        item_text = Text('MCP: Mod Coder Pack', color='#00FFFF') # CYAN hex code
        item_text.scale(0.8)

        # Position the text slightly above the center
        item_text.move_to(UP * 0.5)

        # Define the checkmark
        check = Checkmark(color='#00FF00') # Green hex code
        check.scale(0.8)

        # Position the checkmark to the left of the text
        check.next_to(item_text, LEFT, buff=0.5)

        # Display the text first
        self.play(Write(item_text), run_time=1.5)
        self.wait(0.5)

        # Animate the checkmark appearing
        self.play(Create(check), run_time=1)

        # Hold the final scene
        self.wait(2)
```
